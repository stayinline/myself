# Spark SQL 实战：端到端 ETL、分区写入与 Flink SQL 对比

## 一、端到端 Batch ETL Demo

### 1.1 场景描述

模拟一个典型的数仓 DWD 层加工流程：
- **输入**：CSV 格式的原始订单数据 + Hive 用户维表
- **处理**：清洗 → 关联 → 聚合
- **输出**：Parquet 格式写入 Hive DWS 层

```
┌──────────┐    ┌──────────┐    ┌──────────┐    ┌──────────┐
│ CSV      │    │ Hive     │    │ Spark    │    │ Parquet  │
│ 原始订单  │───►│ 用户维表  │───►│ ETL Job  │───►│ Hive DWS │
│ /data/   │    │ dim_user │    │          │    │ dws_xxx  │
└──────────┘    └──────────┘    └──────────┘    └──────────┘
```

### 1.2 准备测试数据

```python
from pyspark.sql import SparkSession
from pyspark.sql import functions as F
from pyspark.sql.types import *

spark = SparkSession.builder \
    .appName("etl-demo") \
    .master("local[4]") \
    .enableHiveSupport() \
    .config("spark.sql.shuffle.partitions", "8") \
    .config("spark.sql.sources.partitionOverwriteMode", "dynamic") \
    .getOrCreate()

spark.sparkContext.setLogLevel("WARN")

# === 生成模拟 CSV 数据 ===
import os, csv, random
os.makedirs("/tmp/etl_demo/raw_orders", exist_ok=True)

with open("/tmp/etl_demo/raw_orders/orders.csv", "w", newline="") as f:
    writer = csv.writer(f)
    writer.writerow(["order_id", "user_id", "amount", "status", "order_date"])
    for i in range(100000):
        writer.writerow([
            i,
            random.randint(1, 500),
            round(random.uniform(10, 2000), 2),
            random.choice(["paid", "cancelled", "refunded", "pending"]),
            f"2024-{random.randint(1,6):02d}-{random.randint(1,28):02d}"
        ])

# === 创建用户维表（本地 Parquet 模拟 Hive 表）===
users_df = spark.createDataFrame(
    [(uid, f"user_{uid}", random.choice(["北京","上海","广州","深圳"])) 
     for uid in range(1, 501)],
    ["user_id", "user_name", "city"]
)
users_df.write.mode("overwrite").parquet("/tmp/etl_demo/dim_user")
```

### 1.3 DataFrame API 方式实现 ETL

```python
# ========== Step 1: 读取并定义 Schema ==========
order_schema = StructType([
    StructField("order_id", LongType()),
    StructField("user_id", LongType()),
    StructField("amount", DecimalType(10, 2)),
    StructField("status", StringType()),
    StructField("order_date", DateType())
])

raw_orders = (spark.read
    .schema(order_schema)
    .option("header", "true")
    .csv("/tmp/etl_demo/raw_orders/")
)

dim_user = spark.read.parquet("/tmp/etl_demo/dim_user")

print(f"Raw orders count: {raw_orders.count()}")
print(f"Dim user count: {dim_user.count()}")

# ========== Step 2: 数据清洗 ==========
cleaned_orders = (raw_orders
    # 过滤无效记录
    .filter(F.col("order_id").isNotNull())
    .filter(F.col("amount") > 0)
    .filter(F.col("status").isin("paid", "pending"))  # 排除 cancelled/refunded
    # 去重
    .dropDuplicates(["order_id"])
    # 添加清洗标记
    .withColumn("etl_time", F.current_timestamp())
)

print(f"Cleaned orders count: {cleaned_orders.count()}")

# ========== Step 3: 关联维表 ==========
enriched_orders = (cleaned_orders
    .join(dim_user, on="user_id", how="left")
    .withColumn("city", F.coalesce(F.col("city"), F.lit("unknown")))
)

# ========== Step 4: 聚合计算 ==========
daily_summary = (enriched_orders
    .groupBy("order_date", "city")
    .agg(
        F.count("*").alias("order_cnt"),
        F.sum("amount").alias("total_amount"),
        F.avg("amount").alias("avg_amount"),
        F.countDistinct("user_id").alias("distinct_users")
    )
    .orderBy("order_date", "city")
)

# ========== Step 5: 写入 Parquet（按日期分区）==========
(daily_summary
    .write
    .mode("overwrite")
    .partitionBy("order_date")
    .option("compression", "snappy")
    .parquet("/tmp/etl_demo/dws_daily_summary")
)

print("ETL completed! Output written to /tmp/etl_demo/dws_daily_summary")
```

### 1.4 Spark SQL 方式实现同一 ETL

```python
# 注册临时视图
cleaned_orders.createOrReplaceTempView("cleaned_orders")
dim_user.createOrReplaceTempView("dim_user")

# 用纯 SQL 完成关联 + 聚合
result_sql = spark.sql("""
    SELECT 
        o.order_date,
        COALESCE(u.city, 'unknown') AS city,
        COUNT(*) AS order_cnt,
        SUM(o.amount) AS total_amount,
        AVG(o.amount) AS avg_amount,
        COUNT(DISTINCT o.user_id) AS distinct_users
    FROM cleaned_orders o
    LEFT JOIN dim_user u ON o.user_id = u.user_id
    GROUP BY o.order_date, COALESCE(u.city, 'unknown')
    ORDER BY o.order_date, city
""")

# 写入（SQL 方式需要先创建目标表或使用 INSERT OVERWRITE）
result_sql.createOrReplaceTempView("dws_result")
spark.sql("""
    CREATE TABLE IF NOT EXISTS dws_daily_summary_hive
    USING parquet
    PARTITIONED BY (order_date)
    LOCATION '/tmp/etl_demo/dws_daily_summary_hive'
""")

# 动态分区写入
spark.conf.set("spark.sql.sources.partitionOverwriteMode", "dynamic")
spark.sql("""
    INSERT OVERWRITE TABLE dws_daily_summary_hive
    SELECT city, order_cnt, total_amount, avg_amount, distinct_users, order_date
    FROM dws_result
""")
```

### 1.5 两种方式对比

| 维度 | DataFrame API | Spark SQL |
|------|--------------|-----------|
| **可读性** | 链式调用，IDE 补全好 | 类传统 SQL，分析师友好 |
| **类型安全** | 编译期部分检查（Dataset） | 运行时检查 |
| **复用性** | 容易封装函数/类 | 字符串拼接较麻烦 |
| **调试** | 可逐步 show/printSchema | 需 explain 或临时 view |
| **性能** | 完全相同（都走 Catalyst） | 完全相同 |
| **推荐场景** | 工程化 ETL 代码 | Ad-hoc 查询、BI 对接 |

> **面试话术**：DataFrame API 和 Spark SQL 在性能上完全等价，因为最终都经过 Catalyst 优化器。选择哪个取决于团队习惯和项目特点。工程化的 ETL 管道我倾向 DataFrame API，因为有 IDE 补全、单元测试方便、易于封装。Ad-hoc 分析和 BI 报表对接用 SQL 更直接。

## 二、小文件问题

### 2.1 问题表现

```
/user/hive/warehouse/dws_table/dt=2024-06-01/
├── part-00000-xxx.snappy.parquet   (1KB)   ← 太小！
├── part-00001-xxx.snappy.parquet   (2KB)
├── part-00002-xxx.snappy.parquet   (1KB)
...
└── part-00199-xxx.snappy.parquet   (3KB)   ← 200个小文件
```

**危害**：
- NameNode 元数据压力（每个文件占 ~150 bytes 内存）
- 读取时大量 Task 启动开销
- HDFS 块利用率低（默认 128MB 块，实际只用了 1KB）

### 2.2 产生原因

| 原因 | 说明 |
|------|------|
| shuffle partitions 过多 | `spark.sql.shuffle.partitions` 默认 200，小数据集产生 200 个小文件 |
| 动态分区写入 | 每个分区 × 每个 Task = 大量小文件 |
| 流式写入 | 每个 micro-batch 写一批小文件 |
| 频繁 INSERT INTO | 每次追加新文件 |

### 2.3 解决方案

```python
# 方案一：写入前 coalesce/repartition
df.coalesce(10).write.parquet("output")     # 减少分区数
df.repartition(20, "dt").write.parquet("out") # 按分区列重分区

# 方案二：调整 shuffle partitions
spark.conf.set("spark.sql.shuffle.partitions", "20")  # 根据数据量调整

# 方案三：AQE 自动合并（Spark 3.x 推荐）
spark.conf.set("spark.sql.adaptive.enabled", "true")
spark.conf.set("spark.sql.adaptive.coalescePartitions.enabled", "true")
spark.conf.set("spark.sql.adaptive.coalescePartitions.minPartitionSize", "64MB")

# 方案四：事后合并（定期维护任务）
spark.read.parquet("output") \
    .repartition(10) \
    .write.mode("overwrite").parquet("output_compacted")

# 方案五：Hive 层面合并
# ALTER TABLE dws_table CONCATENATE;  -- 仅 ORC
```

### 2.4 AQE 合并原理

```
AQE 检测到某些分区输出 < minPartitionSize
         │
         ▼
将多个小分区的 Shuffle Output 合并为一个分区
         │
         ▼
下游 Stage 的 Task 数减少 → 输出文件数减少

示例：200个分区中180个只有1MB
→ AQE 合并为约 10 个分区
→ 最终输出从 200 个小文件变为 ~30 个合理大小文件
```

## 三、分区写入最佳实践

### 3.1 静态分区 vs 动态分区

```python
# 静态分区：明确指定分区值
df.filter(F.col("dt") == "2024-06-01") \
  .write.mode("overwrite") \
  .insertInto("target_table")  # 只覆盖 dt=2024-06-01

# 动态分区：让 Spark 根据数据中的列值自动分区
spark.conf.set("hive.exec.dynamic.partition", "true")
spark.conf.set("spark.sql.sources.partitionOverwriteMode", "dynamic")

df.write.mode("overwrite") \
  .insertInto("target_table")  # 只覆盖数据中涉及的分区
```

> ⚠️ **重要陷阱**：如果不设置 `partitionOverwriteMode=dynamic`，`INSERT OVERWRITE` 会删除整个表再重写，而非只覆盖涉及的分区！这是生产事故高发点。

### 3.2 写入模式选择

| 模式 | 行为 | 适用场景 |
|------|------|---------|
| `overwrite` | 覆盖目标（全部或动态分区） | 全量刷新 |
| `append` | 追加新文件 | 增量写入 |
| `ignore` | 目标存在则跳过 | 幂等初始化 |
| `errorIfExists` | 目标存在则报错 | 安全检查 |

## 四、Spark SQL vs Flink SQL 批处理体验对比

### 4.1 语法差异

| 特性 | Spark SQL | Flink SQL |
|------|-----------|-----------|
| DDL 语法 | `CREATE TABLE ... USING parquet` | `CREATE TABLE ... WITH ('connector'='filesystem', ...)` |
| 分区声明 | `PARTITIONED BY (col)` | `PARTITIONED BY (col)` + `ALTER TABLE ADD PARTITION` |
| INSERT 语义 | `INSERT OVERWRITE/INTO` | `INSERT INTO/OVERWRITE`（批模式） |
| 时间函数 | `current_timestamp()` | `CURRENT_TIMESTAMP` |
| 窗口函数 | 标准 SQL Window | 标准 + TVF（TUMBLE/HOP/CUMULATE） |
| Catalog | Hive/In-Memory | Hive/Generic/Paimon |
| UDF 注册 | `CREATE FUNCTION ... AS 'class'` | `CREATE FUNCTION ... AS 'class'` |

### 4.2 批处理体验对比

| 维度 | Spark SQL | Flink SQL |
|------|-----------|-----------|
| **Hive 兼容** | ⭐⭐⭐⭐⭐ 原生支持 | ⭐⭐⭐ 通过 HiveCatalog |
| **执行效率** | ⭐⭐⭐⭐⭐ 成熟的 CBO + CodeGen | ⭐⭐⭐ 批模式仍在追赶 |
| **Connector 丰富度** | ⭐⭐⭐⭐ 主流存储全覆盖 | ⭐⭐⭐⭐ CDC/Kafka 更强 |
| **交互式查询** | ⭐⭐⭐⭐ spark-sql/beeline | ⭐⭐ sql-client 较慢 |
| **增量批处理** | ⭐⭐ 有限 | ⭐⭐⭐⭐ Paimon 原生支持 |
| **统一流批语法** | ⭐⭐ SS 有独立语义 | ⭐⭐⭐⭐⭐ 同一套 SQL |

### 4.3 面试话术

> **Q: Spark SQL 和 Flink SQL 做批处理有什么区别？**
>
> A: 纯批处理场景，Spark SQL 仍然是首选。它的 Hive 兼容性最好，CBO 优化最成熟，执行效率高，社区资料也最丰富。Flink SQL 的优势在于"同一套 SQL 既能跑批又能跑流"，如果你的项目既有离线又有实时需求，用 Flink SQL 可以减少维护两套 SQL 的成本。但如果只做离线 ETL 和数仓分析，没必要为了"统一"而牺牲 Spark SQL 的成熟度和性能。
>
> 现在的趋势是用湖表（Paimon/Iceberg）作为中间层，Spark SQL 和 Flink SQL 都能读写同一张表，这样既保留了各自的执行优势，又在存储层实现了统一。

## 五、完整 ETL 脚本模板

```python
#!/usr/bin/env python3
"""
Spark Batch ETL Template
用法: spark-submit --master yarn --deploy-mode cluster etl_template.py
"""

from pyspark.sql import SparkSession
from pyspark.sql import functions as F
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


def create_spark_session(app_name: str) -> SparkSession:
    """创建标准化的 SparkSession"""
    return (SparkSession.builder
        .appName(app_name)
        .enableHiveSupport()
        .config("spark.sql.adaptive.enabled", "true")
        .config("spark.sql.adaptive.coalescePartitions.enabled", "true")
        .config("spark.sql.sources.partitionOverwriteMode", "dynamic")
        .config("spark.sql.shuffle.partitions", "200")
        .getOrCreate()
    )


def extract(spark: SparkSession) -> dict:
    """数据抽取"""
    logger.info("Extracting data...")
    return {
        "orders": spark.read.parquet("hdfs:///data/dwd/orders/"),
        "users": spark.read.parquet("hdfs:///data/dim/users/")
    }


def transform(data: dict) -> "DataFrame":
    """数据转换"""
    logger.info("Transforming data...")
    orders = data["orders"]
    users = data["users"]
    
    return (orders
        .filter(F.col("status") == "paid")
        .join(users, on="user_id", how="left")
        .groupBy("order_date", "city")
        .agg(
            F.count("*").alias("order_cnt"),
            F.sum("amount").alias("total_amount")
        )
    )


def load(df, target_path: str):
    """数据加载"""
    logger.info(f"Loading to {target_path}...")
    (df.write
        .mode("overwrite")
        .partitionBy("order_date")
        .option("compression", "snappy")
        .parquet(target_path)
    )


def main():
    spark = create_spark_session("daily-order-summary")
    try:
        data = extract(spark)
        result = transform(data)
        load(result, "hdfs:///data/dws/daily_order_summary/")
        logger.info("ETL completed successfully!")
    except Exception as e:
        logger.error(f"ETL failed: {e}", exc_info=True)
        raise
    finally:
        spark.stop()


if __name__ == "__main__":
    main()
```
