## 一、Primary Key Model 实时 Upsert/Delete 实操

### 1.1 场景设定

模拟一个电商订单实时数仓场景：
- 订单创建 → INSERT
- 订单支付 → UPDATE status + pay_time
- 订单取消 → DELETE
- 部分字段修正 → UPDATE 单列

### 1.2 建表与初始数据

```sql
-- 创建 PK Model 订单表
CREATE TABLE demo_order_realtime
(
    order_id      BIGINT NOT NULL COMMENT '订单ID',
    user_id       BIGINT NOT NULL COMMENT '用户ID',
    product_id    BIGINT NOT NULL COMMENT '商品ID',
    status        VARCHAR(32) DEFAULT 'created' COMMENT '订单状态',
    amount        DECIMAL(18,2) DEFAULT 0 COMMENT '金额',
    pay_time      DATETIME COMMENT '支付时间',
    ship_time     DATETIME COMMENT '发货时间',
    created_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME DEFAULT CURRENT_TIMESTAMP
)
PRIMARY KEY(order_id)
DISTRIBUTED BY HASH(order_id) BUCKETS 4
PROPERTIES (
    "replication_num" = "1",
    "enable_persistent_index" = "true"
);

-- 批量插入初始订单
INSERT INTO demo_order_realtime
(order_id, user_id, product_id, status, amount, created_at, updated_at)
VALUES
(10001, 501, 2001, 'created', 199.00, NOW(), NOW()),
(10002, 502, 2002, 'created', 399.50, NOW(), NOW()),
(10003, 503, 2003, 'created', 89.90, NOW(), NOW()),
(10004, 504, 2001, 'created', 199.00, NOW(), NOW()),
(10005, 505, 2004, 'created', 599.00, NOW(), NOW());

-- 确认数据
SELECT * FROM demo_order_realtime ORDER BY order_id;
-- 5 行，全部 status='created'
```

### 1.3 Upsert 操作验证

```sql
-- === 场景1: 订单支付（整行 Upsert）===
INSERT INTO demo_order_realtime VALUES
(10001, 501, 2001, 'paid', 199.00, NOW(), NULL, NOW(), NOW());

-- 立即查询验证
SELECT order_id, status, pay_time FROM demo_order_realtime WHERE order_id = 10001;
-- ✅ 返回: 10001 | paid | 2025-06-22 xx:xx:xx
-- 写入即生效，无需等待 Merge

-- === 场景2: 多个订单同时支付 ===
INSERT INTO demo_order_realtime VALUES
(10002, 502, 2002, 'paid', 399.50, NOW(), NULL, NOW(), NOW()),
(10003, 503, 2003, 'paid', 89.90, NOW(), NULL, NOW(), NOW());

-- 验证
SELECT order_id, status FROM demo_order_realtime
WHERE order_id IN (10001, 10002, 10003) ORDER BY order_id;
-- ✅ 三行都是 'paid'
```

### 1.4 部分列更新验证

```sql
-- === 场景3: 只更新发货时间（其他列保持不变）===
UPDATE demo_order_realtime
SET ship_time = NOW(), status = 'shipped', updated_at = NOW()
WHERE order_id = 10001;

-- 验证：amount 和 user_id 等未被更新的列保持原值
SELECT * FROM demo_order_realtime WHERE order_id = 10001;
-- ✅ status='shipped', ship_time 有值, amount=199.00（未变）

-- === 对比 ClickHouse：无法做到部分列更新 ===
-- CH ReplacingMergeTree 必须 INSERT 完整的新版本行
-- 如果漏掉某个列，该列会变成默认值而非保留旧值
```

### 1.5 DELETE 操作验证

```sql
-- === 场景4: 订单取消删除 ===
DELETE FROM demo_order_realtime WHERE order_id = 10004;

-- 立即验证
SELECT count(*) FROM demo_order_realtime WHERE order_id = 10004;
-- ✅ 返回: 0
-- DELETE 立即生效！

-- 条件删除
DELETE FROM demo_order_realtime WHERE user_id = 505 AND status = 'created';
-- 删除 user_505 的未支付订单

-- 验证最终状态
SELECT order_id, user_id, status FROM demo_order_realtime ORDER BY order_id;
-- 10001 | 501 | shipped
-- 10002 | 502 | paid
-- 10003 | 503 | paid
-- (10004 已删除, 10005 已删除)
```

## 二、异步物化视图多表 JOIN 实操

### 2.1 准备基表

```sql
-- 事实表：订单明细
CREATE TABLE fact_orders
(
    order_date   DATE NOT NULL,
    order_id     BIGINT NOT NULL,
    user_id      BIGINT NOT NULL,
    product_id   BIGINT NOT NULL,
    category_id  INT NOT NULL,
    amount       DECIMAL(18,2),
    quantity     INT
)
DUPLICATE KEY(order_date, order_id)
PARTITION BY RANGE(order_date)
(
    PARTITION p20250601 VALUES LESS THAN ('2025-06-02'),
    PARTITION p20250602 VALUES LESS THAN ('2025-06-03'),
    PARTITION p20250603 VALUES LESS THAN ('2025-06-04')
)
DISTRIBUTED BY HASH(order_id) BUCKETS 4
PROPERTIES ("replication_num" = "1");

-- 维表：商品类目
CREATE TABLE dim_category
(
    category_id   INT NOT NULL,
    category_name VARCHAR(128),
    parent_id     INT
)
PRIMARY KEY(category_id)
DISTRIBUTED BY HASH(category_id) BUCKETS 1
PROPERTIES ("replication_num" = "1");

-- 插入测试数据
INSERT INTO dim_category VALUES
(1, '电子产品', 0), (2, '手机', 1), (3, '服装', 0), (4, '食品', 0);

INSERT INTO fact_orders VALUES
('2025-06-01', 1, 101, 1001, 2, 2999.00, 1),
('2025-06-01', 2, 102, 2001, 3, 199.00, 2),
('2025-06-01', 3, 103, 1002, 2, 4999.00, 1),
('2025-06-02', 4, 101, 3001, 4, 59.90, 3),
('2025-06-02', 5, 104, 2002, 3, 299.00, 1),
('2025-06-03', 6, 105, 1003, 2, 1999.00, 1);
```

### 2.2 创建异步 MV

```sql
-- 创建多表 JOIN 预聚合 MV
CREATE MATERIALIZED VIEW mv_daily_category_sales
PARTITION BY (order_date)
DISTRIBUTED BY HASH(order_date) BUCKETS 4
REFRESH ASYNC EVERY (INTERVAL 1 MINUTE)   -- 每分钟刷新（演示用）
AS
SELECT
    f.order_date,
    c.category_name,
    COUNT(DISTINCT f.user_id) AS buyer_cnt,
    COUNT(*) AS order_cnt,
    SUM(f.amount) AS total_amount,
    SUM(f.quantity) AS total_quantity
FROM fact_orders f
JOIN dim_category c ON f.category_id = c.category_id
GROUP BY f.order_date, c.category_name;

-- 查看 MV 状态
SHOW MATERIALIZED VIEWS WHERE NAME = 'mv_daily_category_sales'\G
-- 等待 last_refresh_state = SUCCESS

-- 手动触发首次刷新（不等定时）
REFRESH MATERIALIZED VIEW mv_daily_category_sales;
```

### 2.3 验证透明改写

```sql
-- 开启透明改写
SET enable_materialized_view_rewrite = true;

-- 执行原始 JOIN 查询
EXPLAIN SELECT
    f.order_date,
    c.category_name,
    COUNT(DISTINCT f.user_id) AS buyer_cnt,
    SUM(f.amount) AS total_amount
FROM fact_orders f
JOIN dim_category c ON f.category_id = c.category_id
GROUP BY f.order_date, c.category_name;

-- EXPLAIN 输出中应包含:
-- OlapScanNode: table=mv_daily_category_sales  ← 命中 MV!
-- 而非原始的 fact_orders + dim_category JOIN

-- 实际查询结果
SELECT
    f.order_date,
    c.category_name,
    COUNT(DISTINCT f.user_id) AS buyer_cnt,
    SUM(f.amount) AS total_amount
FROM fact_orders f
JOIN dim_category c ON f.category_id = c.category_id
GROUP BY f.order_date, c.category_name
ORDER BY f.order_date, c.category_name;
```

### 2.4 增量刷新验证

```sql
-- 向新分区写入数据
INSERT INTO fact_orders VALUES
('2025-06-03', 7, 106, 3002, 4, 129.00, 2),
('2025-06-03', 8, 107, 2003, 3, 459.00, 1);

-- 等待 MV 自动刷新（1分钟），或手动触发
REFRESH MATERIALIZED VIEW mv_daily_category_sales;

-- 查看刷新记录，确认只刷新了变更分区

SELECT * FROM mv_daily_category_sales
WHERE order_date = '2025-06-03'
ORDER BY category_name;
```

## 三、ClickHouse vs StarRocks 同查询对比

### 3.1 对比实验设计

使用相同的数据集和业务查询，分别在两个引擎上执行并记录指标。

```
测试数据集:
- 事实表: 1亿行订单记录
- 维表: 1万行商品信息
- 查询: 按日期+品类统计 GMV、订单数、买家数

硬件配置:
- 3节点, 每节点 16C 64GB NVMe SSD
- 两引擎使用相同硬件
```

### 3.2 查询对比

```sql
-- === ClickHouse 版本 ===
-- 需要提前把维表 JOIN 打成宽表（CH JOIN 弱）
-- 或者接受慢速 JOIN

-- 方案A: 宽表直接查（ETL 预处理）
SELECT
    order_date,
    category_name,
    uniqExact(user_id) AS buyer_cnt,
    count() AS order_cnt,
    sum(amount) AS total_amount
FROM wide_order_table
WHERE order_date BETWEEN '2025-06-01' AND '2025-06-30'
GROUP BY order_date, category_name
ORDER BY order_date, category_name;
-- 冷查: ~800ms, 热查: ~200ms ✅ 快但需要维护宽表

-- 方案B: 运行时 JOIN（不推荐）
SELECT ... FROM orders a JOIN products b ON a.product_id = b.id ...
-- 冷查: ~15s ❌ 极慢

-- === StarRocks 版本 ===
-- 直接写标准 SQL，CBO 自动优化
SELECT
    f.order_date,
    c.category_name,
    COUNT(DISTINCT f.user_id) AS buyer_cnt,
    COUNT(*) AS order_cnt,
    SUM(f.amount) AS total_amount
FROM fact_orders f
JOIN dim_category c ON f.category_id = c.category_id
WHERE f.order_date BETWEEN '2025-06-01' AND '2025-06-30'
GROUP BY f.order_date, c.category_name
ORDER BY f.order_date, c.category_name;
-- 冷查: ~500ms, 热查: ~100ms ✅ 无需宽表，原生 JOIN 就够快

-- === StarRocks + 异步 MV ===
-- 如果创建了 mv_daily_category_sales，透明改写后：
-- 冷查: ~50ms, 热查: ~10ms ✅✅ 命中 MV 后极速
```

### 3.3 对比总结

| 指标 | CH 宽表 | CH JOIN | SR 原生 JOIN | SR + Async MV |
|------|---------|---------|-------------|---------------|
| 冷查延迟 | ~800ms | ~15s | ~500ms | ~50ms |
| 热查延迟 | ~200ms | ~12s | ~100ms | ~10ms |
| ETL 复杂度 | 高（需打宽表） | 低 | 低 | 低（MV 自动维护） |
| SQL 灵活性 | 低（受限于宽表维度） | 高 | 高 | 高（透明改写） |
| 数据新鲜度 | 取决于 ETL 频率 | 实时 | 实时 | 取决于 MV 刷新间隔 |
| 开发体验 | 差 | 差（性能不可控） | 好 | 最好 |

## 四、Routine Load 消费 Kafka 实操

### 4.1 前置准备

```bash
# 确保 Kafka 集群可用，创建 Topic
kafka-topics.sh --create \
  --topic order_events \
  --partitions 6 \
  --replication-factor 3 \
  --bootstrap-server kafka1:9092

# 向 Topic 发送测试消息
echo '{"order_date":"2025-06-22","order_id":20001,"user_id":601,"product_id":3001,"amount":299.00,"status":"created"}' | \
  kafka-console-producer.sh --topic order_events --bootstrap-server kafka1:9092
```

### 4.2 创建 Routine Load 任务

```sql
-- 目标表
CREATE TABLE order_events_kafka
(
    order_date   DATE,
    order_id     BIGINT,
    user_id      BIGINT,
    product_id   BIGINT,
    amount       DECIMAL(18,2),
    status       VARCHAR(32)
)
DUPLICATE KEY(order_date, order_id)
PARTITION BY RANGE(order_date) ()
DISTRIBUTED BY HASH(order_id) BUCKETS 4
PROPERTIES (
    "replication_num" = "1",
    "dynamic_partition.enable" = "true",
    "dynamic_partition.time_unit" = "DAY",
    "dynamic_partition.start" = "-7",
    "dynamic_partition.end" = "3",
    "dynamic_partition.prefix" = "p"
);

-- 创建 Routine Load
CREATE ROUTINE LOAD rl_order_events ON order_events_kafka
COLUMNS(order_date, order_id, user_id, product_id, amount, status)
PROPERTIES (
    "desired_concurrent_number" = "3",
    "max_batch_interval" = "10",
    "max_batch_rows" = "100000",
    "max_batch_size" = "104857600"
)
FROM KAFKA (
    "kafka_broker_list" = "kafka1:9092,kafka2:9092,kafka3:9092",
    "kafka_topic" = "order_events",
    "property.group.id" = "sr_rl_order_group",
    "property.auto.offset.reset" = "earliest",
    "property.kafka_default_offsets" = "OFFSET_BEGINNING"
);
```

### 4.3 监控与排障

```sql
-- 查看消费状态
SHOW ROUTINE LOAD FOR rl_order_events\G
/*
关注字段:
  State: RUNNING / PAUSED / STOPPED
  Progress: 各 Partition 消费到的 Offset
  Lag: 消费延迟（落后最新 Offset 的消息数）
  ErrorLogUrls: 错误日志 URL（如果有脏数据）
  LatestSourcePosition: Kafka 最新 Offset
*/

-- 如果 State=PAUSED，查看暂停原因
SHOW ROUTINE LOAD TASK WHERE JobName = 'rl_order_events';

-- 常见暂停原因及处理:
-- 1. Kafka 不可达 → 检查网络和 Broker 状态
-- 2. 脏数据过多 → 设置 strict_mode=false 或增加 max_filter_ratio
-- 3. BE 负载过高 → 降低 desired_concurrent_number

-- 恢复暂停的任务
RESUME ROUTINE LOAD FOR rl_order_events;

-- 修改消费参数（不停任务）
ALTER ROUTINE LOAD FOR rl_order_events
PROPERTIES (
    "desired_concurrent_number" = "6",
    "max_batch_interval" = "5"
);
```

## 五、面试话术

### Q: 你们怎么保证实时数据的准确性和时效性？

> 我们用 Flink Connector 写入 StarRocks Primary Key Model，Flink Checkpoint 保证 Exactly-Once 语义。PK Model 的 Merge-on-Write 机制保证写入即可见，不存在 ClickHouse ReplacingMergeTree 那种需要等 Merge 的问题。实测从 Kafka 消息产生到 StarRocks 可查，端到端延迟在 10 秒以内。对于偶尔的数据修正需求，直接用 UPDATE/DELETE 语句就能处理，不需要像 ClickHouse 那样走重量级的 Mutation。

### Q: 异步物化视图在实际项目中怎么用？

> 我们用它来加速高频的多表 JOIN 查询。比如运营看板需要按日期×品类×地区统计 GMV，涉及事实表 JOIN 三个维表。直接在 StarRocks 上跑这个 JOIN 大约 500ms，加了异步 MV 后降到 20ms。MV 每 5 分钟刷新一次，对运营看板来说完全够用。最大的好处是应用层不需要改代码，CBO 会自动识别并改写到 MV，这对已有系统的升级非常友好。
