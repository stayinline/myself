## 一、本地/Docker 环境搭建

### 1.1 Docker Compose 快速启动（推荐 allin1 单容器）

```yaml
# docker-compose-starrocks.yml
version: '3.8'

services:
  starrocks:
    image: starrocks/allin1-ubuntu:3.3.0
    container_name: starrocks
    hostname: starrocks
    ports:
      - "8030:8030"   # FE HTTP
      - "9030:9030"   # MySQL Protocol
      - "8040:8040"   # BE HTTP
    environment:
      - TZ=Asia/Shanghai
    volumes:
      - sr-data:/data/deploy/starrocks
    healthcheck:
      test: ["CMD", "mysql", "-h", "127.0.0.1", "-P", "9030", "-u", "root", "-e", "SHOW BACKENDS"]
      interval: 10s
      timeout: 5s
      retries: 15

volumes:
  sr-data:
```

> allin1 镜像已内置 FE+BE，**无需**再 `ADD BACKEND`。生产环境请用官方多节点部署。

### 1.2 初始化与连接

```bash
# 启动集群
docker compose -f docker-compose-starrocks.yml up -d

# 等待健康检查通过后连接
mysql -h 127.0.0.1 -P 9030 -u root -e "SHOW FRONTENDS\G; SHOW BACKENDS\G"
# 确认 Alive=true
```

### 1.3 使用 MySQL 客户端连接

```bash
# StarRocks 完全兼容 MySQL 协议
mysql -h 127.0.0.1 -P 9030 -u root
```

## 二、创建四种表模型体验差异

### 2.1 Duplicate Model（明细表）

```sql
-- 创建明细事件表
CREATE TABLE events_duplicate
(
    event_time  DATETIME NOT NULL COMMENT '事件时间',
    user_id     BIGINT NOT NULL COMMENT '用户ID',
    session_id  VARCHAR(64) COMMENT '会话ID',
    page_url    VARCHAR(1024) COMMENT '页面URL',
    action_type VARCHAR(32) COMMENT '动作类型',
    duration_ms INT COMMENT '停留时长(ms)',
    device      VARCHAR(32) COMMENT '设备类型',
    ip          VARCHAR(45) COMMENT 'IP地址'
)
DUPLICATE KEY(event_time, user_id)
PARTITION BY RANGE(event_time)
(
    PARTITION p202506 VALUES LESS THAN ('2025-07-01'),
    PARTITION p202507 VALUES LESS THAN ('2025-08-01')
)
DISTRIBUTED BY HASH(user_id) BUCKETS 4
PROPERTIES (
    "replication_num" = "1",           -- 单机测试设为1
    "dynamic_partition.enable" = "true",
    "dynamic_partition.time_unit" = "MONTH",
    "dynamic_partition.start" = "-3",
    "dynamic_partition.end" = "1",
    "dynamic_partition.prefix" = "p"
);

-- 插入测试数据
INSERT INTO events_duplicate VALUES
('2025-06-01 10:00:00', 1001, 'sess_001', '/home', 'view', 5000, 'mobile', '192.168.1.1'),
('2025-06-01 10:00:05', 1001, 'sess_001', '/product/123', 'click', 200, 'mobile', '192.168.1.1'),
('2025-06-01 10:01:00', 1002, 'sess_002', '/home', 'view', 8000, 'desktop', '10.0.0.1'),
('2025-06-01 10:01:30', 1001, 'sess_001', '/checkout', 'view', 3000, 'mobile', '192.168.1.1');

-- 验证：所有行都保留
SELECT * FROM events_duplicate ORDER BY event_time;
-- 返回 4 行，包括同一用户的多次访问
```

### 2.2 Aggregate Model（聚合表）

```sql
-- 创建预聚合表
CREATE TABLE page_metrics_aggregate
(
    stat_date   DATE NOT NULL COMMENT '统计日期',
    page_url    VARCHAR(1024) NOT NULL COMMENT '页面URL',
    pv          SUM REPLACE DEFAULT "0" COMMENT 'PV计数',
    uv_bitmap   BITMAP_UNION BITMAP_DEFAULT "" COMMENT 'UV位图',
    max_dur     MAX REPLACE DEFAULT "0" COMMENT '最大停留时长',
    min_dur     MIN REPLACE DEFAULT "0" COMMENT '最小停留时长'
)
AGGREGATE KEY(stat_date, page_url)
PARTITION BY RANGE(stat_date)
(
    PARTITION p202506 VALUES LESS THAN ('2025-07-01')
)
DISTRIBUTED BY HASH(page_url) BUCKETS 4
PROPERTIES ("replication_num" = "1");

-- 分批写入相同 Key 的数据，观察自动聚合
INSERT INTO page_metrics_aggregate VALUES
('2025-06-01', '/home', 1, to_bitmap(1001), 5000, 5000),
('2025-06-01', '/home', 1, to_bitmap(1002), 8000, 8000);

INSERT INTO page_metrics_aggregate VALUES
('2025-06-01', '/home', 1, to_bitmap(1003), 3000, 3000);

-- 触发 Compaction 确保合并
ADMIN EXECUTE ON (ALL BACKENDS) "COMPACT TABLE page_metrics_aggregate";

-- 查询：Aggregate Model 直接读取聚合列（无需再 GROUP BY 聚合列）
SELECT
    stat_date,
    page_url,
    pv,
    bitmap_union_count(uv_bitmap) AS uv,
    max_dur,
    min_dur
FROM page_metrics_aggregate;

-- 预期结果: 2025-06-01 | /home | 3 | 3 | 8000 | 3000
-- PV=3 (1+1+1), UV=3 (1001∪1002∪1003), Max=8000, Min=3000
```

### 2.3 Unique Model（唯一键表）

```sql
CREATE TABLE user_profile_unique
(
    user_id     BIGINT NOT NULL,
    nickname    VARCHAR(128),
    age         INT,
    city        VARCHAR(64),
    level       VARCHAR(32),
    updated_at  DATETIME
)
UNIQUE KEY(user_id)
DISTRIBUTED BY HASH(user_id) BUCKETS 4
PROPERTIES (
    "replication_num" = "1",
    "enable_unique_key_merge_on_write" = "true"  -- 启用 MoW 模式
);

-- 初始写入
INSERT INTO user_profile_unique VALUES
(1001, 'Alice', 25, 'Beijing', 'silver', '2025-06-01 10:00:00'),
(1002, 'Bob', 30, 'Shanghai', 'gold', '2025-06-01 10:00:00');

-- Upsert: 更新 Alice 的城市和等级
INSERT INTO user_profile_unique VALUES
(1001, 'Alice', 25, 'Hangzhou', 'gold', '2025-06-01 11:00:00');

-- 验证：Alice 的信息已更新（整行覆盖）
SELECT * FROM user_profile_unique WHERE user_id = 1001;
-- 返回: 1001 | Alice | 25 | Hangzhou | gold | 2025-06-01 11:00:00

-- ⚠️ Unique Model 不支持 DELETE（请用 Primary Key Model）
-- ⚠️ 不支持部分列更新，Upsert 必须提供完整行
```

### 2.4 Primary Key Model（主键表）⭐

```sql
CREATE TABLE order_realtime_pk
(
    order_id    BIGINT NOT NULL,
    user_id     BIGINT NOT NULL,
    status      VARCHAR(32) DEFAULT 'created',
    amount      DECIMAL(18,2) DEFAULT 0,
    pay_time    DATETIME,
    updated_at  DATETIME DEFAULT CURRENT_TIMESTAMP
)
PRIMARY KEY(order_id)
DISTRIBUTED BY HASH(order_id) BUCKETS 4
PROPERTIES (
    "replication_num" = "1",
    "enable_persistent_index" = "true"
);

-- 初始写入
INSERT INTO order_realtime_pk VALUES
(10001, 1001, 'created', 99.90, NULL, NOW()),
(10002, 1002, 'created', 199.00, NULL, NOW());

-- ✅ Upsert: 更新订单状态
INSERT INTO order_realtime_pk VALUES
(10001, 1001, 'paid', 99.90, NOW(), NOW());

-- ✅ 部分列更新: 只更新 status 列（其他列保持原值）
UPDATE order_realtime_pk SET status = 'shipped' WHERE order_id = 10001;

-- ✅ DELETE: 删除订单
DELETE FROM order_realtime_pk WHERE order_id = 10002;

-- 验证最终状态
SELECT * FROM order_realtime_pk ORDER BY order_id;
-- 返回:
-- 10001 | 1001 | shipped | 99.90 | <pay_time> | <updated_at>
-- (10002 已被删除，不再出现)
```

## 三、Primary Key Model vs ClickHouse ReplacingMergeTree 对比实验

### 3.1 对比维度

| 对比项 | StarRocks PK Model | CH ReplacingMergeTree |
|--------|-------------------|----------------------|
| UPDATE 语法 | INSERT/UPDATE 原生支持 | 只能 INSERT 新版本 + argMax/FINAL |
| DELETE 语法 | DELETE FROM ... WHERE ... | ALTER TABLE DELETE (重量级 Mutation) |
| 部分列更新 | ✅ UPDATE SET col=val | ❌ 不支持 |
| 实时可见性 | 写入即生效 | 需等后台 Merge 或 FINAL |
| 查询一致性 | 强一致 | 最终一致 |
| 写入吞吐 | ~15万 rows/s | ~30万 rows/s |
| 内存开销 | 较高（主键索引） | 较低 |

### 3.2 实操对比

```sql
-- === StarRocks 侧 ===
-- 写入后立即查到最新值
INSERT INTO order_realtime_pk VALUES (10001, 1001, 'paid', 99.90, NOW(), NOW());
SELECT status FROM order_realtime_pk WHERE order_id = 10001;
-- 立即返回: paid ✅

-- DELETE 立即生效
DELETE FROM order_realtime_pk WHERE order_id = 10001;
SELECT count(*) FROM order_realtime_pk WHERE order_id = 10001;
-- 立即返回: 0 ✅

-- === ClickHouse 侧 ===
-- 写入后旧版本仍然存在
INSERT INTO orders_replacing VALUES (10001, 'created', 1);
INSERT INTO orders_replacing VALUES (10001, 'paid', 2);
SELECT status FROM orders_replacing WHERE id = 10001;
-- 可能返回两行: created, paid ⚠️

-- 必须用 FINAL 或 argMax
SELECT argMax(status, version) FROM orders_replacing WHERE id = 10001 GROUP BY id;
-- 返回: paid ✅ 但性能差

-- DELETE 是异步 Mutation
ALTER TABLE orders_replacing DELETE WHERE id = 10001;
-- 不立即生效！需要等后台执行
SELECT count(*) FROM orders_replacing WHERE id = 10001;
-- 可能仍返回 > 0 ⚠️
```

## 四、Bucket/Partition 设计考量

### 4.1 Partition 设计原则

```
┌──────────────────────────────────────────────────────┐
│              Partition 设计决策树                       │
│                                                      │
│  数据是否有明确的时间维度？                              │
│  ├── 是 → RANGE Partition by Date/DateTime            │
│  │   • 按天分区：日增数据 > 1GB                         │
│  │   • 按月分区：日增数据 < 1GB                         │
│  │   • 动态分区：避免手动维护                            │
│  └── 否 → LIST Partition by 枚举字段                   │
│      • 如 region, tenant_id 等                        │
│                                                      │
│  分区数量控制：                                         │
│  • 单表分区数建议 < 1000                               │
│  • 每个分区至少包含一个 Tablet                           │
│  • 分区过细 → 元数据膨胀，Compaction 碎片化              │
│  • 分区过粗 → 无法有效裁剪，TTL 粒度不够                 │
└──────────────────────────────────────────────────────┘
```

### 4.2 Bucket 设计原则

```
┌──────────────────────────────────────────────────────┐
│              Bucket 设计公式                            │
│                                                      │
│  Bucket 数 ≈ 单个 Tablet 理想大小 / 预估分区大小         │
│                                                      │
│  推荐单个 Tablet 大小: 1GB ~ 10GB                      │
│                                                      │
│  示例：                                                │
│  • 分区大小 50GB → Bucket = 50/5 = 10                  │
│  • 分区大小 5GB  → Bucket = 5/1 = 5                    │
│  • 分区大小 500MB → Bucket = 1（最小值）                 │
│                                                      │
│  注意事项：                                             │
│  • Bucket 数一旦创建不可修改                             │
│  • Distribution Key 选高基数字段（如 user_id）           │
│  • 避免低基数导致数据倾斜                                │
│  • 如果需要 Colocate Join，多表 Bucket 数和 Key 要一致   │
│  • 3.x 支持 Auto Bucket，可让系统自动决定               │
└──────────────────────────────────────────────────────┘
```

### 4.3 实际建表示例

```sql
-- 大型事实表：日增 10GB，保留 2 年
CREATE TABLE fact_order
(
    order_date  DATE NOT NULL,
    order_id    BIGINT NOT NULL,
    user_id     BIGINT NOT NULL,
    amount      DECIMAL(18,2),
    status      VARCHAR(32)
)
PRIMARY KEY(order_id)
PARTITION BY RANGE(order_date) ()
DISTRIBUTED BY HASH(order_id) BUCKETS AUTO   -- 3.x Auto Bucket
PROPERTIES (
    "replication_num" = "3",
    "dynamic_partition.enable" = "true",
    "dynamic_partition.time_unit" = "DAY",
    "dynamic_partition.start" = "-730",
    "dynamic_partition.end" = "3",
    "dynamic_partition.prefix" = "p",
    "enable_persistent_index" = "true"
);

-- 小型维度表：总量 < 1GB
CREATE TABLE dim_product
(
    product_id  BIGINT NOT NULL,
    name        VARCHAR(256),
    category    VARCHAR(64),
    price       DECIMAL(18,2)
)
PRIMARY KEY(product_id)
DISTRIBUTED BY HASH(product_id) BUCKETS 1    -- 小表 1 个 Bucket 足够
PROPERTIES ("replication_num" = "3");
```

## 五、常见踩坑与排查

### 5.1 新手常见问题

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| 建表失败 "no alive backend" | BE 未添加或未启动 | `ALTER SYSTEM ADD BACKEND` |
| 查询很慢 | Bucket 过多/过少 | 调整 Bucket 数，参考 4.2 |
| 写入报错 "tablet not found" | 分区不存在 | 检查动态分区或手动添加分区 |
| PK Model 写入慢 | 主键过大 | 使用 enable_persistent_index |
| Compaction 跟不上 | 高频小批量写入 | 调大 batch size，调整 compaction 参数 |
| 内存溢出 | 大查询无 LIMIT | 加 LIMIT / 调整 query_mem_limit |

## 六、面试话术

### Q: StarRocks 的四种表模型你都了解吗？实际怎么选的？

> 了解。我们项目中四种都用到了。日志和埋点数据用 Duplicate Model，纯追加不需要去重。报表预聚合层用 Aggregate Model，写入时自动 SUM/COUNT。用户画像维度表用 Unique Model，CDC 同步过来只需要简单的 Key 去重。核心的实时订单事实表用 Primary Key Model，因为需要支持订单状态的实时更新和部分字段修正，还偶尔需要 DELETE 处理退款取消的场景。PK Model 的优势是写入即可见、查询强一致，代价是写入吞吐比 Duplicate 低约 60%，我们通过攒批写入来缓解。

### Q: Bucket 数量怎么确定？

> 核心原则是让每个 Tablet 的大小在 1-10GB 之间。计算公式是 Bucket 数 ≈ 分区大小 / 目标 Tablet 大小。比如日分区 50GB，目标 5GB 一个 Tablet，就设 10 个 Bucket。StarRocks 3.x 还支持 Auto Bucket，可以让系统根据数据量自动调整，适合数据量波动大的场景。另外要注意 Distribution Key 必须选高基数字段，否则会导致数据倾斜。如果两张表需要做 Colocate Join，它们的 Distribution Key 和 Bucket 数必须完全一致。
