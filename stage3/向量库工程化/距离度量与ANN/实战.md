# 距离度量与 ANN 实战

## 一、用 NumPy 手算三种距离

### 1.1 基础实现

```python
import numpy as np

def euclidean_distance(a: np.ndarray, b: np.ndarray) -> float:
    """L2 欧氏距离"""
    return np.sqrt(np.sum((a - b) ** 2))

def inner_product(a: np.ndarray, b: np.ndarray) -> float:
    """内积（点积），越大越相似"""
    return np.dot(a, b)

def cosine_similarity(a: np.ndarray, b: np.ndarray) -> float:
    """余弦相似度，越大越相似"""
    norm_a = np.linalg.norm(a)
    norm_b = np.linalg.norm(b)
    if norm_a == 0 or norm_b == 0:
        return 0.0
    return np.dot(a, b) / (norm_a * norm_b)

def cosine_distance(a: np.ndarray, b: np.ndarray) -> float:
    """余弦距离，越小越相似"""
    return 1.0 - cosine_similarity(a, b)
```

### 1.2 验证归一化后三者等价

```python
np.random.seed(42)
D = 768
N = 10000

# 生成随机向量并归一化
vectors = np.random.randn(N, D).astype(np.float32)
vectors = vectors / np.linalg.norm(vectors, axis=1, keepdims=True)
query = np.random.randn(D).astype(np.float32)
query = query / np.linalg.norm(query)

# 分别用三种度量计算 Top-10
dist_l2 = np.array([euclidean_distance(query, v) for v in vectors])
dist_ip = np.array([-inner_product(query, v) for v in vectors])  # 取负使"越小越近"
dist_cos = np.array([cosine_distance(query, v) for v in vectors])

top10_l2 = np.argsort(dist_l2)[:10]
top10_ip = np.argsort(dist_ip)[:10]
top10_cos = np.argsort(dist_cos)[:10]

print(f"L2 Top-10 IDs:     {sorted(top10_l2)}")
print(f"IP Top-10 IDs:     {sorted(top10_ip)}")
print(f"Cosine Top-10 IDs: {sorted(top10_cos)}")
print(f"三者完全一致: {set(top10_l2) == set(top10_ip) == set(top10_cos)}")
# 输出: 三者完全一致: True
```

### 1.3 未归一化时三者不等价

```python
# 不归一化
raw_vectors = np.random.randn(N, D).astype(np.float32)
raw_query = np.random.randn(D).astype(np.float32)

dist_l2_raw = np.array([euclidean_distance(raw_query, v) for v in raw_vectors])
dist_ip_raw = np.array([-inner_product(raw_query, v) for v in raw_vectors])
dist_cos_raw = np.array([cosine_distance(raw_query, v) for v in raw_vectors])

top10_l2_raw = np.argsort(dist_l2_raw)[:10]
top10_ip_raw = np.argsort(dist_ip_raw)[:10]
top10_cos_raw = np.argsort(dist_cos_raw)[:10]

print(f"未归一化 L2 vs IP 一致: {set(top10_l2_raw) == set(top10_ip_raw)}")       # False
print(f"未归一化 L2 vs Cosine 一致: {set(top10_l2_raw) == set(top10_cos_raw)}") # False
print(f"未归一化 IP vs Cosine 一致: {set(top10_ip_raw) == set(top10_cos_raw)}") # False
```

> ⚠️ **关键发现**：未归一化时三种度量的 Top-K 结果完全不同！这就是为什么必须确认你的 Embedding 模型是否输出了归一化向量。

## 二、精确搜索 vs ANN 搜索对比实验

### 2.1 使用 FAISS 进行对比

```python
import faiss
import time
import numpy as np

# 准备数据
D = 768
N = 1_000_000  # 100万向量
NQ = 100       # 100个查询

np.random.seed(42)
xb = np.random.randn(N, D).astype(np.float32)
xb = xb / np.linalg.norm(xb, axis=1, keepdims=True)  # 归一化

xq = np.random.randn(NQ, D).astype(np.float32)
xq = xq / np.linalg.norm(xq, axis=1, keepdims=True)

K = 10  # Top-K

# ========== 精确搜索 (Flat IP) ==========
index_flat = faiss.IndexFlatIP(D)
index_flat.add(xb)

start = time.perf_counter()
D_flat, I_flat = index_flat.search(xq, K)
flat_time = time.perf_counter() - start
print(f"Flat IP: {flat_time:.3f}s, QPS={NQ/flat_time:.1f}")

# ========== HNSW 搜索 ==========
M = 16
efConstruction = 200
index_hnsw = faiss.IndexHNSWFlat(D, M, faiss.METRIC_INNER_PRODUCT)
index_hnsw.hnsw.efConstruction = efConstruction
index_hnsw.add(xb)

# 测试不同 efSearch
for ef_search in [16, 32, 64, 128, 256]:
    index_hnsw.hnsw.efSearch = ef_search
    
    start = time.perf_counter()
    D_hnsw, I_hnsw = index_hnsw.search(xq, K)
    hnsw_time = time.perf_counter() - start
    
    # 计算 Recall@K
    recall = 0.0
    for i in range(NQ):
        recall += len(set(I_flat[i]) & set(I_hnsw[i])) / K
    recall /= NQ
    
    print(f"HNSW efSearch={ef_search:>3d}: {hnsw_time:.3f}s, "
          f"QPS={NQ/hnsw_time:.1f}, Recall@{K}={recall:.4f}")
```

### 2.2 预期输出示例

```
Flat IP: 2.847s, QPS=35.1
HNSW efSearch= 16: 0.012s, QPS=8333.3, Recall@10=0.8760
HNSW efSearch= 32: 0.021s, QPS=4761.9, Recall@10=0.9340
HNSW efSearch= 64: 0.039s, QPS=2564.1, Recall@10=0.9720
HNSW efSearch=128: 0.074s, QPS=1351.4, Recall@10=0.9910
HNSW efSearch=256: 0.142s, QPS=704.2,  Recall@10=0.9980
```

### 2.3 分析

| efSearch | Recall@10 | QPS | 延迟(ms) | 适用场景 |
|----------|-----------|-----|---------|---------|
| 16 | 0.876 | 8333 | 0.12 | 探索性搜索、推荐粗排 |
| 32 | 0.934 | 4762 | 0.21 | 一般推荐系统 |
| 64 | 0.972 | 2564 | 0.39 | RAG / 知识问答 |
| 128 | 0.991 | 1351 | 0.74 | 高精度 RAG |
| 256 | 0.998 | 704 | 1.42 | 去重 / 安全审核 |

> 可以看到 efSearch 从 16 → 256，Recall 从 87.6% 提升到 99.8%，但 QPS 下降了约 12 倍。**没有最优参数，只有最适合业务的参数**。

## 三、实际场景的模型→距离配对实践

### 3.1 OpenAI Embeddings

```python
from openai import OpenAI
import numpy as np

client = OpenAI()

def get_openai_embedding(text: str) -> np.ndarray:
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=text
    )
    vec = np.array(response.data[0].embedding, dtype=np.float32)
    
    # 验证是否归一化
    norm = np.linalg.norm(vec)
    print(f"向量范数: {norm:.6f}")  # 应该接近 1.0
    
    return vec

# OpenAI 的向量已归一化 → 用 IP 或 Cosine 均可
# FAISS 中推荐 IndexFlatIP 或 IndexHNSWFlat(D, M, METRIC_INNER_PRODUCT)
```

### 3.2 BGE-M3

```python
from FlagEmbedding import BGEM3FlagModel
import numpy as np

model = BGEM3FlagModel('BAAI/bge-m3', use_fp16=True)

texts = ["机器学习是人工智能的子领域", "深度学习属于机器学习的分支"]
output = model.encode(texts, return_dense=True)
vecs = output['dense_vecs']

# 验证归一化
norms = np.linalg.norm(vecs, axis=1)
print(f"向量范数: {norms}")  # [1.0, 1.0]

# BGE-M3 归一化 → 用 IP/Cosine
# 注意：BGE-M3 还支持 sparse embedding 和 ColBERT，那些是不同的检索范式
```

## 四、常见陷阱与避坑指南

### 陷阱 1：未归一化向量直接用 Inner Product

```python
# ❌ 错误：模长大的向量会被过度偏好
a = np.array([100, 0, 0])  # 模长 100
b = np.array([0.9, 0.1, 0.1])  # 模长 ≈ 0.91
c = np.array([1, 0, 0])  # 模长 1

query = np.array([1, 0, 0])

print(f"IP(q,a) = {np.dot(query, a)}")   # 100 ← 因为模长大而被选中
print(f"IP(q,b) = {np.dot(query, b)}")   # 0.9
print(f"IP(q,c) = {np.dot(query, c)}")   # 1

# 但从语义方向上看，c 和 query 完全一致，a 只是碰巧模长大
# ✅ 正确：先归一化再用 IP，或者直接用 L2/Cosine
```

### 陷阱 2：混合不同模型的向量

```python
# ❌ 绝对不能这样做
openai_vec = get_openai_embedding("hello world")  # 1536维
bge_vec = get_bge_embedding("hello world")         # 1024维

# 即使维度相同也不行！不同模型的向量空间完全不兼容
# 同一个句子在不同模型中的向量表示没有任何可比性

# ✅ 同一索引中的所有向量必须来自同一模型、同一版本
```

### 陷阱 3：忽略维度灾难的影响

```python
# 在低维数据上 KD-Tree 很好用
from sklearn.neighbors import KDTree

low_dim_data = np.random.randn(100000, 10)
tree = KDTree(low_dim_data)
# 搜索很快 ✓

# 但在高维 Embedding 上完全失效
high_dim_data = np.random.randn(100000, 768)
tree_hd = KDTree(high_dim_data)
# 搜索退化为接近暴力扫描 ✗

# ✅ 高维数据请使用 HNSW / IVF 等专为高维设计的索引
```

### 陷阱 4：Recall 评估用了错误的 Ground Truth

```python
# ❌ 不要用另一个 ANN 索引的结果当 ground truth
# 比如用 HNSW efSearch=64 的结果评估 HNSW efSearch=128

# ✅ 必须用 Flat Index（暴力搜索）的结果作为 ground truth
index_gt = faiss.IndexFlatIP(D)
index_gt.add(xb)
D_gt, I_gt = index_gt.search(xq, K)  # 这才是真正的 Top-K

# 然后用这个 I_gt 来计算任何 ANN 索引的 Recall
```

## 五、面试话术

### Q："向量检索为什么要用近似？精度和速度怎么权衡？"

> **为什么要近似**：十亿级向量做暴力搜索，单次查询要遍历所有向量，计算量是 O(N×D)。以 10 亿 × 768 维为例，单次查询就是 7680 亿次浮点运算，加上内存带宽瓶颈，延迟在数十秒级别，在线服务根本不可接受。ANN 通过构建索引结构把搜索范围缩小到一小部分候选集，将复杂度降到 O(logN) 级别，延迟从秒级降到毫秒级。
>
> **怎么权衡**：我们用 Recall@K 来量化"近似"的质量——拿暴力搜索的 Top-K 作为标准答案，看 ANN 结果命中了多少。每个 ANN 算法都有可调参数来控制这个权衡：HNSW 调 efSearch，IVF 调 nprobe。参数越大，Recall 越高但延迟也越高。实际操作是先确定业务可接受的最低 Recall（比如 RAG ≥ 0.95），然后在这个约束下找延迟最小的参数配置。我们会画一条 Recall-QPS 曲线，选 Pareto 最优点。

### Q："Cosine Similarity 和 Dot Product 在你的项目中用的哪个？为什么？"

> 我们用的是 Dot Product（内积）。因为我们使用的 BGE-M3 模型输出的向量已经是 L2 归一化的，对归一化向量来说 IP 和 Cosine 排序完全等价。选 IP 的原因是 FAISS 和 Milvus 对 IP 有更好的底层优化（SIMD 指令集加速），而且避免了每次查询时重复做归一化的开销。如果用的是未归一化的模型比如原始 CLIP，那我会选择 L2 距离或者在入库前统一做归一化。

### Q："你怎么确认你的向量确实是归一化的？"

> 两个层面验证。第一，在模型接入时做一次离线检查：采样几千条数据过模型，计算每条向量的 L2 范数，确认都在 1.0 ± 1e-5 范围内。第二，在写入向量库之前加一道断言/校验，如果范数偏离阈值就告警。另外也可以看模型文档——OpenAI、BGE、E5 这些主流模型都明确说明了输出是否归一化。对于不确定的模型，宁可多一步归一化也不要假设它已经归一化了。

## 六、完整实验脚本模板

```python
"""
distance_benchmark.py — 距离度量 + ANN 综合评测脚本
用法: python distance_benchmark.py --dim 768 --num 1000000 --queries 100
"""
import argparse
import numpy as np
import faiss
import time
import json

def run_benchmark(D, N, NQ, K=10):
    np.random.seed(42)
    xb = np.random.randn(N, D).astype(np.float32)
    xb /= np.linalg.norm(xb, axis=1, keepdims=True)
    xq = np.random.randn(NQ, D).astype(np.float32)
    xq /= np.linalg.norm(xq, axis=1, keepdims=True)

    results = {}

    # Ground Truth
    gt = faiss.IndexFlatIP(D)
    gt.add(xb)
    _, I_gt = gt.search(xq, K)
    
    # HNSW sweep
    for M in [16, 32]:
        for efC in [100, 200]:
            idx = faiss.IndexHNSWFlat(D, M, faiss.METRIC_INNER_PRODUCT)
            idx.hnsw.efConstruction = efC
            idx.add(xb)
            
            for efS in [16, 32, 64, 128, 256, 512]:
                idx.hnsw.efSearch = efS
                t0 = time.perf_counter()
                _, I_ann = idx.search(xq, K)
                dt = time.perf_counter() - t0
                
                recall = np.mean([
                    len(set(I_gt[i]) & set(I_ann[i])) / K 
                    for i in range(NQ)
                ])
                
                key = f"HNSW_M{M}_efC{efC}_efS{efS}"
                results[key] = {
                    "recall": round(recall, 4),
                    "latency_ms": round(dt / NQ * 1000, 2),
                    "qps": round(NQ / dt, 1)
                }
    
    return results

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--dim", type=int, default=768)
    parser.add_argument("--num", type=int, default=1_000_000)
    parser.add_argument("--queries", type=int, default=100)
    args = parser.parse_args()
    
    results = run_benchmark(args.dim, args.num, args.queries)
    print(json.dumps(results, indent=2))
```

这个脚本可以直接运行，产出结构化的 benchmark 结果，用于汇报或调参参考。
