## 一、构建标准评测集

### 1.1 完整评测集构建脚本

```python
import numpy as np
import faiss
import json
import time
from dataclasses import dataclass, asdict
from typing import List, Dict, Tuple

@dataclass
class BenchmarkConfig:
    name: str
    dim: int
    num_db: int
    num_query: int
    k: int
    seed: int = 42

def build_ground_truth(cfg: BenchmarkConfig) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
    """
    构建评测数据集和 Ground Truth
    Returns: (db_vectors, query_vectors, gt_ids)
    """
    np.random.seed(cfg.seed)
    
    # 生成数据库向量（归一化）
    db_vecs = np.random.randn(cfg.num_db, cfg.dim).astype(np.float32)
    db_vecs /= np.linalg.norm(db_vecs, axis=1, keepdims=True)
    
    # 生成查询向量（归一化）
    query_vecs = np.random.randn(cfg.num_query, cfg.dim).astype(np.float32)
    query_vecs /= np.linalg.norm(query_vecs, axis=1, keepdims=True)
    
    # Flat Index 精确搜索作为 Ground Truth
    print(f"Building ground truth (Flat IP, N={cfg.num_db}, D={cfg.dim})...")
    gt_index = faiss.IndexFlatIP(cfg.dim)
    gt_index.add(db_vecs)
    
    t0 = time.perf_counter()
    _, gt_ids = gt_index.search(query_vecs, cfg.k)
    gt_time = time.perf_counter() - t0
    print(f"Ground truth built in {gt_time:.1f}s")
    
    return db_vecs, query_vecs, gt_ids

# 使用示例
cfg = BenchmarkConfig(
    name="benchmark_1M_768",
    dim=768,
    num_db=1_000_000,
    num_query=500,
    k=10
)
db_vecs, query_vecs, gt_ids = build_ground_truth(cfg)
```

## 二、Recall@K 计算实现

### 2.1 标准 Recall 计算

```python
def compute_recall(gt_ids: np.ndarray, ann_ids: np.ndarray, k: int) -> float:
    """
    计算 Recall@K
    
    Args:
        gt_ids: Ground Truth IDs, shape (num_query, k)
        ann_ids: ANN search result IDs, shape (num_query, k)
        k: Top-K value
    
    Returns:
        Mean Recall@K across all queries
    """
    assert gt_ids.shape == ann_ids.shape
    num_query = gt_ids.shape[0]
    
    recalls = []
    for i in range(num_query):
        gt_set = set(gt_ids[i].tolist())
        ann_set = set(ann_ids[i].tolist())
        intersection = len(gt_set & ann_set)
        recalls.append(intersection / k)
    
    return float(np.mean(recalls))

def compute_recall_at_multiple_k(gt_ids_full: np.ndarray, 
                                   ann_ids: np.ndarray, 
                                   k_values: List[int]) -> Dict[int, float]:
    """计算多个 K 值的 Recall"""
    results = {}
    for k in k_values:
        results[k] = compute_recall(
            gt_ids_full[:, :k], 
            ann_ids[:, :k], 
            k
        )
    return results
```

### 2.2 向量化高效计算

```python
def compute_recall_vectorized(gt_ids: np.ndarray, ann_ids: np.ndarray, k: int) -> float:
    """
    向量化版本，比循环快 10-100x
    利用排序+交集的技巧
    """
    # 对每行排序
    gt_sorted = np.sort(gt_ids[:, :k], axis=1)
    ann_sorted = np.sort(ann_ids[:, :k], axis=1)
    
    # 逐行计算交集大小
    intersections = np.zeros(gt_sorted.shape[0], dtype=np.int32)
    for i in range(gt_sorted.shape[0]):
        intersections[i] = len(np.intersect1d(gt_sorted[i], ann_sorted[i]))
    
    return float(np.mean(intersections / k))

# 更快的版本（适合大批量）
def compute_recall_fast(gt_ids: np.ndarray, ann_ids: np.ndarray, k: int) -> float:
    """使用集合操作的快速版本"""
    num_query = gt_ids.shape[0]
    total_hits = 0
    
    # 将每行转为 frozenset 做交集
    for i in range(num_query):
        total_hits += len(set(gt_ids[i, :k]) & set(ann_ids[i, :k]))
    
    return total_hits / (num_query * k)
```

## 三、综合 Benchmark 框架

### 3.1 完整的评测 Runner

```python
import faiss
import numpy as np
import time
from typing import Dict, List, Any

class VectorIndexBenchmark:
    def __init__(self, db_vecs, query_vecs, gt_ids, k=10, n_warmup=10, n_runs=20):
        self.db_vecs = db_vecs
        self.query_vecs = query_vecs
        self.gt_ids = gt_ids
        self.k = k
        self.n_warmup = n_warmup
        self.n_runs = n_runs
        self.D = db_vecs.shape[1]
        self.N = db_vecs.shape[0]
        self.NQ = query_vecs.shape[0]
    
    def benchmark_index(self, index, name: str, params: dict = None) -> dict:
        """对单个索引配置运行完整 benchmark"""
        if params:
            for key, val in params.items():
                parts = key.split(".")
                obj = index
                for p in parts[:-1]:
                    obj = getattr(obj, p)
                setattr(obj, parts[-1], val)
        
        # Warmup
        index.search(self.query_vecs[:min(self.n_warmup, self.NQ)], self.k)
        
        # Benchmark runs
        latencies = []
        all_ann_ids = None
        
        for run in range(self.n_runs):
            t0 = time.perf_counter()
            _, ann_ids = index.search(self.query_vecs, self.k)
            dt = time.perf_counter() - t0
            latencies.append(dt)
            
            if run == 0:
                all_ann_ids = ann_ids
        
        # Compute metrics
        recall = compute_recall(self.gt_ids, all_ann_ids, self.k)
        avg_latency = np.mean(latencies) / self.NQ * 1000  # ms
        p50_latency = np.percentile(latencies, 50) / self.NQ * 1000
        p99_latency = np.percentile(latencies, 99) / self.NQ * 1000
        qps = self.NQ / np.mean(latencies)
        
        result = {
            "name": name,
            "recall@K": round(recall, 4),
            "avg_latency_ms": round(avg_latency, 3),
            "p50_latency_ms": round(p50_latency, 3),
            "p99_latency_ms": round(p99_latency, 3),
            "qps": round(qps, 1),
        }
        
        print(f"  {name:>30s} | Recall={recall:.4f} | "
              f"Avg={avg_latency:.2f}ms | P99={p99_latency:.2f}ms | QPS={qps:.0f}")
        
        return result
    
    def run_comparison(self, configs: List[dict]) -> List[dict]:
        """运行多组配置的对比实验"""
        results = []
        
        for cfg in configs:
            idx_type = cfg["type"]
            name = cfg["name"]
            params = cfg.get("params", {})
            
            if idx_type == "flat":
                index = faiss.IndexFlatIP(self.D)
                index.add(self.db_vecs)
                
            elif idx_type == "hnsw":
                M = cfg.get("M", 16)
                efC = cfg.get("efConstruction", 200)
                index = faiss.IndexHNSWFlat(self.D, M, faiss.METRIC_INNER_PRODUCT)
                index.hnsw.efConstruction = efC
                index.add(self.db_vecs)
                
            elif idx_type == "ivf_flat":
                nlist = cfg.get("nlist", 4096)
                quantizer = faiss.IndexFlatIP(self.D)
                index = faiss.IndexIVFFlat(quantizer, self.D, nlist, 
                                           faiss.METRIC_INNER_PRODUCT)
                index.train(self.db_vecs)
                index.add(self.db_vecs)
                
            elif idx_type == "ivf_pq":
                nlist = cfg.get("nlist", 4096)
                m = cfg.get("m", 96)
                nbits = cfg.get("nbits", 8)
                quantizer = faiss.IndexFlatIP(self.D)
                index = faiss.IndexIVFPQ(quantizer, self.D, nlist, m, nbits,
                                         faiss.METRIC_INNER_PRODUCT)
                index.train(self.db_vecs)
                index.add(self.db_vecs)
            
            else:
                raise ValueError(f"Unknown index type: {idx_type}")
            
            result = self.benchmark_index(index, name, params)
            results.append(result)
        
        return results
```

recall_monitor.py — 定期采样验证线上 Recall
可以作为 CronJob 或独立服务运行

import numpy as np
import faiss
from pymilvus import connections, Collection
import prometheus_client

# Prometheus gauge
recall_gauge = prometheus_client.Gauge(
    'vector_search_recall', 
    'Sampled recall@10 of online vector search',
    ['collection', 'index_type']
)

def monitor_recall(collection_name, sample_queries, gt_ids, k=10):
    """采样验证线上 Recall"""
    collection = Collection(collection_name)
    
    search_params = {"metric_type": "IP", "params": {"ef": 64}}
    _, ann_ids = collection.search(
        data=sample_queries.tolist(),
        anns_field="embedding",
        param=search_params,
        limit=k,
        consistency_level="Eventually"
    )
    
    recall = compute_recall(gt_ids, np.array(ann_ids), k)
    
    # 上报到 Prometheus
    recall_gauge.labels(
        collection=collection_name, 
        index_type="HNSW"
    ).set(recall)
    
    print(f"[{datetime.now()}] Recall@{k}={recall:.4f} for {collection_name}")
    return recall

## 六、面试话术

### Q："你怎么做向量检索的评测？评测集怎么构建？"

> 我们的评测分三层。第一层是离线的算法级评测：用 Flat Index 暴力搜索的结果作为 Ground Truth，构建了包含 500 条查询的标准评测集。查询是从真实用户日志中抽样并去重的，确保覆盖各种查询模式。数据库向量也是从生产数据中随机采样的 100 万条。对每种索引配置跑 Recall@10/Recall@100、P99 延迟、QPS，画出 Pareto 曲线来选最优参数。
>
> 第二层是端到端的业务评测：我们有一套人工标注的 golden set，每条查询有专家标注的相关文档列表。用 NDCG@10 评估排序质量，同时看 RAG 的回答准确率。这层的指标更贴近业务价值。
>
> 第三层是线上持续监控：每周自动跑一次 Recall 回归测试（用固定的评测集在线上环境跑），结果写入 Grafana。如果 Recall 下降超过 1% 就告警，可能是数据分布漂移或索引退化。同时实时监控搜索 P99 延迟和 QPS，这些指标能在秒级反映系统健康状态。

### Q："Delete 之后空间什么时候释放？Compaction 怎么做？"

> 向量数据库的 delete 是软删除——只在 bitset 中标记，不立即释放物理空间。真正的空间回收发生在 Compaction 过程中，Compaction 会合并多个 Segment 并跳过已删除的向量，生成一个干净的新 Segment。
>
> 在我们的实践中，Compaction 有两种触发方式。一是自动触发：Milvus 会在删除比例超过阈值（我们设为 25%）或小 Segment 数量过多时自动触发。二是手动触发：我们在每天凌晨低峰期通过定时任务调用 compaction API，主动整理碎片。
>
> 需要注意的是 Compaction 本身是资源密集操作——它需要读取旧 Segment、合并数据、重建索引。所以我们把它安排在低峰期，并且限制了并发度避免影响在线服务。另外 Compaction 完成后旧的 Segment 会被标记为 dropped，但物理文件不会立即删除，有一个 GC 延迟窗口用于保证一致性。

### Q："你的评测发现过什么意外情况吗？怎么处理的？"

> 有几个印象深刻的案例。第一个是我们发现 IVF-PQ 在某些查询上 Recall 特别低（低于 0.8），而平均 Recall 是 0.93。排查后发现是因为训练数据和实际查询分布有偏差——训练数据是三个月前的，但最近新增了一个品类导致向量分布发生了变化。重新用最新数据训练后 Recall 恢复到了 0.95。教训是训练数据要定期更新。
>
> 第二个是 HNSW 在大量删除后 Recall 下降。因为被删除的节点仍然在图中占据位置，搜索时会命中这些无效节点然后被过滤掉，浪费了 efSearch 的预算。解决方案是在删除比例达到 20% 时触发 Compaction + 索引重建。
>
> 第三个是不同 batch size 下 Recall 不一致。后来发现是因为 Growing Segment 中的数据没有建索引，不同时间点的 Growing Segment 大小不同导致搜索结果有波动。固定了 flush 策略后问题消除。这些都是评测体系帮我们发现的——如果没有标准化的评测和持续的监控，这些问题可能在生产中很久才会被发现。

## 七、评测 Checklist

```
□ Ground Truth 是否使用了 Flat Index（而非另一个 ANN）？
□ 查询集是否有代表性（来自真实日志，非纯随机）？
□ 是否测试了多个 K 值（@10, @50, @100）？
□ 是否关注了 P99/P999 延迟而非仅平均值？
□ 是否在相同硬件条件下做对比？
□ 是否考虑了 warmup 的影响？
□ 是否验证了不同数据规模下的表现？
□ 是否有持续的线上 Recall 监控？
□ 是否在数据分布变化后重新评估？
□ 评测结果是否有版本化和可追溯性？
```
