## 一、实战项目：给订单查询 Agent 加上"长期记忆"

### 1.1 需求场景

```
场景 1: 用户偏好记忆
  第1次对话: "我是张三，以后回答请简洁一点"
  第2次对话: "查下我的订单" → Agent 记得用简洁风格 + 知道是张三

场景 2: 历史交互记忆
  上周: "帮我查了订单O001，发现物流异常"
  今天: "上次那个物流问题解决了吗？" → Agent 记得 O001 和物流异常

场景 3: 领域知识积累
  管理员: "新增规则：VIP用户订单优先处理"
  后续: Agent 在处理 VIP 用户时自动应用此规则
```

### 1.4 记忆管理器实现

```python
import time
import numpy as np
from typing import List, Optional

class MemoryManager:
    """Agent 长期记忆管理器"""

    MEMORY_TYPES = ["preference", "fact", "task_result", "correction", "rule"]

    def __init__(self, embedding_model, vector_store, llm_client):
        self.embedder = embedding_model
        self.store = vector_store
        self.llm = llm_client

    def store_memory(self, user_id: str, content: str,
                     memory_type: str, confidence: float = 0.8):
        """存储一条新记忆"""
        if memory_type not in self.MEMORY_TYPES:
            raise ValueError(f"Invalid memory type: {memory_type}")

        embedding = self.embedder.encode(content)
        now = int(time.time())

        self.store.insert({
            "embedding": [embedding],
            "content": [content],
            "user_id": [user_id],
            "memory_type": [memory_type],
            "created_at": [now],
            "confidence": [confidence],
            "access_count": [0],
            "last_accessed_at": [now],
        })
        self.store.flush()

    def retrieve_memories(self, user_id: str, query: str,
                          top_k: int = 5, min_score: float = 0.6,
                          memory_types: Optional[List[str]] = None) -> List[dict]:
        """检索相关记忆，含时间衰减和相关性过滤"""
        query_embedding = self.embedder.encode(query)
        now = int(time.time())

        # 构建过滤表达式
        filters = [f'user_id == "{user_id}"', f'confidence > {min_score}']
        if memory_types:
            type_filter = " OR ".join(
                [f'memory_type == "{t}"' for t in memory_types]
            )
            filters.append(f"({type_filter})")

        search_expr = " AND ".join(filters)

        results = self.store.search(
            data=[query_embedding],
            anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"ef": 128}},
            limit=top_k * 2,  # 多取一些用于重排序
            expr=search_expr,
            output_fields=["content", "memory_type", "created_at",
                           "confidence", "access_count"]
        )

        # 时间衰减重排序
        scored_results = []
        for hit in results[0]:
            age_hours = (now - hit.entity.get("created_at")) / 3600
            time_decay = max(0.3, 1.0 / (1 + age_hours / 720))  # 30天半衰期

            final_score = hit.score * 0.7 + time_decay * 0.2 + \
                         hit.entity.get("confidence") * 0.1

            scored_results.append({
                "content": hit.entity.get("content"),
                "memory_type": hit.entity.get("memory_type"),
                "score": final_score,
                "raw_similarity": hit.score,
                "age_hours": age_hours,
            })

        # 按最终分数排序并截取 top_k
        scored_results.sort(key=lambda x: x["score"], reverse=True)
        return scored_results[:top_k]

    def extract_and_store(self, user_id: str, conversation: List[dict]):
        """从对话中自动提取值得记忆的信息"""
        extract_prompt = f"""分析以下对话，提取值得长期记住的信息。

对话内容：
{self._format_conversation(conversation)}

请以 JSON 数组返回，每项包含：
- content: 要记住的内容（简洁明确）
- memory_type: preference/fact/task_result/correction/rule
- confidence: 确信度 0.0-1.0

如果没有值得记住的内容，返回空数组 []。
只提取明确的、有长期价值的信息，不要过度提取。"""

        response = self.llm.generate(extract_prompt)
        memories = parse_json(response)

        for mem in memories:
            self.store_memory(
                user_id=user_id,
                content=mem["content"],
                memory_type=mem["memory_type"],
                confidence=mem.get("confidence", 0.7)
            )

        return len(memories)
```

### 1.5 将记忆注入 Agent Prompt

```python
MEMORY_INJECTION_TEMPLATE = """
## 用户历史记忆（按相关性排序）

{memory_block}

注意：
- 以上记忆来自历史交互，可能不完全准确
- 如果当前信息与记忆冲突，以当前信息为准
- 不要主动提及"我记得..."除非用户问起
"""

def build_memory_augmented_prompt(user_id, query, memory_manager):
    """构建带记忆的增强 prompt"""
    memories = memory_manager.retrieve_memories(
        user_id=user_id,
        query=query,
        top_k=5,
        min_score=0.5
    )

    if not memories:
        return ""  # 无相关记忆，不注入

    memory_lines = []
    for i, mem in enumerate(memories, 1):
        type_label = {
            "preference": "偏好",
            "fact": "事实",
            "task_result": "历史结果",
            "correction": "纠正",
            "rule": "规则"
        }.get(mem["memory_type"], "其他")

        memory_lines.append(
            f"{i}. [{type_label}] {mem['content']} (相关度: {mem['score']:.2f})"
        )

    return MEMORY_INJECTION_TEMPLATE.format(
        memory_block="\n".join(memory_lines)
    )
```

## 二、Multi-Agent 复杂度体验

### 2.1 简单 Multi-Agent 示例：代码生成 + Review

```python
class CodeGenReviewSystem:
    """双 Agent 系统：生成 + 审查"""

    def __init__(self, llm_client):
        self.llm = llm_client
        self.cost_tracker = {"total_tokens": 0, "total_calls": 0}

    def generate_and_review(self, requirement: str, max_rounds: int = 3):
        code = None
        review = None

        for round_num in range(max_rounds):
            print(f"\n=== Round {round_num + 1} ===")

            # Step 1: Generator 生成/修改代码
            gen_prompt = f"""你是一个资深Python开发者。
{'根据以下需求编写代码：' if code is None else '根据以下review意见修改代码：'}
{'需求：' + requirement if code is None else ''}
{'当前代码：\\n' + code if code else ''}
{'Review意见：\\n' + review if review else ''}

请直接输出完整代码，不要解释。"""

            code = self.llm.generate(gen_prompt)
            self.cost_tracker["total_calls"] += 1
            print(f"[Generator] 生成了 {len(code)} 字符的代码")

            # Step 2: Reviewer 审查代码
            review_prompt = f"""你是一个严格的Code Reviewer。
审查以下代码的正确性、安全性和可维护性。

需求：{requirement}
代码：
{code}

如果有问题，列出具体问题和建议。
如果代码没问题，回复 "APPROVED"。"""

            review = self.llm.generate(review_prompt)
            self.cost_tracker["total_calls"] += 1
            print(f"[Reviewer] {review[:100]}...")

            if "APPROVED" in review:
                print(f"\n✅ 通过审查！共 {round_num + 1} 轮")
                break

        print(f"\n📊 统计: {self.cost_tracker['total_calls']} 次 LLM 调用")
        return code, review
```

### 2.2 成本爆炸实测数据

```
单 Agent 直接生成代码:
  - LLM 调用次数: 1
  - Token 消耗: ~3K
  - 耗时: ~5s
  - 代码质量: 7/10

Multi-Agent (Gen + Review, 2轮通过):
  - LLM 调用次数: 4 (2 gen + 2 review)
  - Token 消耗: ~12K (4x)
  - 耗时: ~20s (4x)
  - 代码质量: 9/10

Multi-Agent (Gen + Review, 3轮通过):
  - LLM 调用次数: 6
  - Token 消耗: ~20K (6.7x)
  - 耗时: ~30s (6x)
  - 代码质量: 9.5/10

结论: 质量提升 20-30%，成本增加 400-600%
      ROI 取决于场景对质量的要求
```

### 2.3 Multi-Agent 常见坑

| 坑 | 表现 | 解决方案 |
|----|------|----------|
| 无限辩论 | Reviewer 总能找到问题，永远不 APPROVED | 设置 max_rounds + 强制通过条件 |
| 信息丢失 | Agent 间传递时关键细节被遗漏 | 结构化消息格式 + 完整性校验 |
| 角色混淆 | Generator 开始做 Review 的事 | 强化 system prompt 的角色边界 |
| 一致性差 | 每次运行结果差异大 | Temperature=0 + 固定 seed |
| 调试地狱 | 不知道哪个 Agent 出了问题 | 全链路 trace + 每步日志 |
| 死锁 | A 等 B，B 等 A | 超时机制 + 依赖图检测 |

## 三、记忆检索相关性挑战

### 3.1 典型失败案例

```
案例 1: 时效性问题
  记忆库中有: "用户使用 Java 8" (6个月前)
  用户现在问: "帮我配置开发环境"
  检索返回: Java 8 的配置（过时了）
  正确做法: 应该询问当前使用的版本

案例 2: 语义歧义
  用户问: "怎么删除缓存？"
  记忆库中有: "不要随意删除生产缓存" (安全警告)
  检索返回: 这条安全警告（语义相关但不是操作指南）
  正确做法: 区分"操作方法"和"注意事项"两种记忆类型

案例 3: 过度召回
  用户问: "张三的订单"
  检索返回 20 条关于"张三"的记忆，包括无关的地址、偏好等
  正确做法: 结合 memory_type 过滤 + 更精确的查询改写
```

### 3.2 提升检索质量的工程手段

```python
class EnhancedMemoryRetrieval:
    """增强版记忆检索"""

    def retrieve(self, query, user_id, context=None):
        # Step 1: 查询改写
        rewritten_query = self.rewrite_query(query, context)

        # Step 2: 多路召回
        semantic_results = self.semantic_search(rewritten_query, user_id)
        keyword_results = self.keyword_search(rewritten_query, user_id)
        recent_results = self.recent_memories(user_id, limit=3)

        # Step 3: 融合排序 (RRF - Reciprocal Rank Fusion)
        fused = self.rrf_fusion([semantic_results, keyword_results, recent_results])

        # Step 4: LLM 精排 (可选，增加延迟但提升精度)
        reranked = self.llm_rerank(query, fused[:10])

        # Step 5: 去重 + 截断
        final = self.deduplicate_and_truncate(reranked, max_items=5)

        return final

    def rrf_fusion(self, result_lists, k=60):
        """Reciprocal Rank Fusion 多路召回融合"""
        scores = {}
        for results in result_lists:
            for rank, item in enumerate(results):
                key = item["content"]
                scores[key] = scores.get(key, 0) + 1.0 / (k + rank + 1)

        sorted_items = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        return [{"content": k, "rrf_score": s} for k, s in sorted_items]
```

## 四、面试话术："Agent 怎么记住历史/长期知识"

### 4.1 推荐回答

> "Agent 的记忆我分三层来做：
> 
> 第一层是工作记忆，就是 Context Window 里的当前对话，零延迟但容量有限；
> 
> 第二层是短期记忆，当对话超过窗口限制时，我用摘要压缩把早期对话浓缩成关键事实摘要，保留最近 3-5 轮完整对话，这样在成本和上下文连贯性之间取得平衡；
> 
> 第三层是长期记忆，本质上就是 RAG——我把历史交互中提取出的关键信息（用户偏好、重要事实、任务结果等）做 embedding 存入 Vector DB，新请求来时做相似度检索回填到 prompt。这里我会加时间衰减权重避免召回过时信息，加 memory_type 过滤提高精准度，加置信度评分防止记忆污染。
> 
> 其实长期记忆这块和我之前做的向量检索基础设施是完全复用的技术栈，Milvus/pgvector + text2vec + HNSW 索引，只是应用场景从文档问答变成了 Agent 记忆。"

### 4.2 进阶追问准备

**Q: 怎么决定什么该存、什么不该存？**
> "我用了两层策略：第一层是规则过滤，只有特定类型的信息才触发存储（用户明确表达的偏好、纠正、关键事实等）；第二层是 LLM 判断，在对话结束后让模型评估这段对话是否有长期价值。关键是宁缺毋滥——存错了比没存更有害，因为错误记忆会持续影响后续交互。"

**Q: 记忆越来越多，怎么处理？**
> "三个机制：一是 TTL，低置信度或长时间未被访问的记忆自动过期；二是合并，相似记忆定期聚合为一条更精炼的记忆；三是重要性评分，高频被访问的记忆提升优先级。这和数据库的冷热分层思路类似。"

**Q: Multi-Agent 和单 Agent 怎么选？**
> "我的原则是能单 Agent 就不拆多 Agent。只有当单个 Agent 的 prompt 过长且角色混杂、工具过多导致选择准确率下降、或者确实需要对抗验证时才考虑拆分。Multi-Agent 的成本通常是 1.5-3 倍，还要面对通信一致性和调试困难的问题。在实际项目中，我更多是用单 Agent + 好的 workflow 编排来解决问题。"
