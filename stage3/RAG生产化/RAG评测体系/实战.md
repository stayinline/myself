# RAG 评测体系 — 实战篇

## 一、构建 QA 评测集

### 1.1 评测集设计原则

| 原则 | 说明 |
|------|------|
| 覆盖度 | 涵盖各类查询类型（事实/对比/聚合/模糊） |
| 真实性 | 来自真实用户查询日志，非凭空编造 |
| 可验证 | 每条都有明确的 ground truth 和相关 chunk ID |
| 多样性 | 难度分布合理（简单/中等/困难） |
| 规模 | 最少 20 条，建议 50+，上不封顶 |

## 二、自动化评测脚本

### 2.1 完整评测管线

```python
"""rag_evaluator.py - RAG 全链路评测"""
import time
import numpy as np
from typing import List, Dict, Optional
from openai import OpenAI

client = OpenAI()


class RAGEvaluator:
    def __init__(self, rag_system, eval_dataset: List[Dict]):
        self.rag = rag_system
        self.dataset = eval_dataset

    def run_evaluation(self, top_k: int = 5) -> Dict:
        """运行完整评测"""
        retrieval_metrics = []
        generation_metrics = []
        latencies = []

        for item in self.dataset:
            start = time.time()

            # 执行 RAG 问答
            result = self.rag.ask(item["question"], top_k=top_k)
            latency = time.time() - start
            latencies.append(latency)

            # 检索侧评测
            ret_metric = self._eval_retrieval(
                retrieved_ids=[c.get("chunk_id", c.get("index")) for c in result["retrieved"]],
                relevant_ids=item["relevant_chunk_ids"],
            )
            retrieval_metrics.append(ret_metric)

            # 生成侧评测
            gen_metric = self._eval_generation(
                question=item["question"],
                answer=result["answer"],
                contexts=[c["content"] for c in result["retrieved"]],
                ground_truth=item["ground_truth"],
            )
            generation_metrics.append(gen_metric)

        # 汇总
        report = {
            "total_questions": len(self.dataset),
            "retrieval": {
                "hit_rate@k": np.mean([m["hit"] for m in retrieval_metrics]),
                "recall@k": np.mean([m["recall"] for m in retrieval_metrics]),
                "mrr": np.mean([m["mrr"] for m in retrieval_metrics]),
            },
            "generation": {
                "faithfulness": np.mean([m["faithfulness"] for m in generation_metrics]),
                "answer_relevancy": np.mean([m["answer_relevancy"] for m in generation_metrics]),
                "context_precision": np.mean([m["context_precision"] for m in generation_metrics]),
            },
            "performance": {
                "avg_latency_s": np.mean(latencies),
                "p95_latency_s": np.percentile(latencies, 95),
                "p99_latency_s": np.percentile(latencies, 99),
            },
        }
        return report

    def _eval_retrieval(self, retrieved_ids: list, relevant_ids: list) -> Dict:
        """计算单条检索指标"""
        retrieved_set = set(str(x) for x in retrieved_ids)
        relevant_set = set(str(x) for x in relevant_ids)

        # Hit Rate: 至少一个命中
        hit = bool(retrieved_set & relevant_set)

        # Recall: 命中数 / 总相关数
        recall = len(retrieved_set & relevant_set) / max(len(relevant_set), 1)

        # MRR: 第一个命中的排名倒数
        mrr = 0.0
        for rank, rid in enumerate(retrieved_ids, 1):
            if str(rid) in relevant_set:
                mrr = 1.0 / rank
                break

        return {"hit": hit, "recall": recall, "mrr": mrr}

    def _eval_generation(self, question, answer, contexts, ground_truth) -> Dict:
        """LLM-as-Judge 评估生成质量"""
        prompt = f"""请评估以下 RAG 回答的质量，每项打 0-1 分。

【问题】{question}
【回答】{answer}
【参考资料】{' | '.join(contexts[:3])}...
【标准答案】{ground_truth}

请以 JSON 输出:
{{
  "faithfulness": <0-1, 回答是否忠实于参考资料>,
  "answer_relevancy": <0-1, 回答是否与问题相关>,
  "context_precision": <0-1, 参考资料中有多少是真正有用的>
}}

只输出 JSON，不要其他内容。"""

        resp = client.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.0,
            response_format={"type": "json_object"},
        )
        import json
        scores = json.loads(resp.choices[0].message.content)
        return scores
```

### 2.2 使用 RAGAS 简化评测

```python
"""eval_with_ragas.py"""
from ragas import evaluate
from ragas.metrics import faithfulness, answer_relevancy, context_precision, context_recall
from datasets import Dataset


def run_ragas_eval(rag_system, eval_dataset):
    """使用 RAGAS 框架评测"""
    # 准备数据
    questions = [item["question"] for item in eval_dataset]
    ground_truths = [item["ground_truth"] for item in eval_dataset]

    # 批量获取 RAG 结果
    answers = []
    contexts_list = []
    for q in questions:
        result = rag_system.ask(q)
        answers.append(result["answer"])
        contexts_list.append([c["content"] for c in result["retrieved"]])

    dataset = Dataset.from_dict({
        "question": questions,
        "answer": answers,
        "contexts": contexts_list,
        "ground_truth": ground_truths,
    })

    results = evaluate(
        dataset,
        metrics=[faithfulness, answer_relevancy, context_precision, context_recall],
    )
    return dict(results)
```

### 4.3 Prometheus + Grafana 集成

```python
"""metrics_exporter.py"""
from prometheus_client import Counter, Histogram, Gauge, start_http_server

# 定义指标
rag_query_total = Counter("rag_query_total", "Total RAG queries", ["status"])
rag_retrieval_hit = Gauge("rag_retrieval_hit_rate", "Retrieval hit rate")
rag_faithfulness = Gauge("rag_faithfulness_score", "Faithfulness score")
rag_latency = Histogram("rag_latency_seconds", "RAG query latency",
                         buckets=[0.5, 1.0, 2.0, 3.0, 5.0, 10.0])
rag_citation_fabricated = Counter("rag_citation_fabricated_total",
                                    "Fabricated citations detected")

# 在 RAG 管线中埋点
def ask_with_metrics(rag, query):
    with rag_latency.time():
        result = rag.ask(query)

    rag_query_total.labels(status="success").inc()
    rag_retrieval_hit.set(result["debug"]["retrieval_hit"])
    rag_faithfulness.set(result["debug"]["faithfulness"])

    if result["debug"].get("fabricated_citations", 0) > 0:
        rag_citation_fabricated.inc(result["debug"]["fabricated_citations"])

    return result

# 启动 metrics server
start_http_server(8000)
```

## 五、面试话术

### Q: "你怎么衡量一个 RAG 系统好不好、怎么持续优化？"

> 我建了一套完整的评测体系，分为**离线评测**和**在线监控**两部分。
>
> **离线评测**方面，我维护了一个 50 条的 QA 评测集，每条都标注了 ground truth 和相关 chunk ID。检索侧看 Recall@5 和 MRR，生成侧用 RAGAS 框架评 Faithfulness、Answer Relevancy 和 Context Precision。每次改完任何组件——不管是 chunking、embedding、检索策略还是 prompt——都跑一遍评测，记录到实验追踪表里，确认指标有提升才上线。
>
> **在线监控**方面，我把关键指标接入了 Prometheus + Grafana：检索命中率、平均延迟、引用编造率、用户负反馈率等。设了告警阈值，比如检索命中率低于 0.6 就触发告警。同时每天抽样做 Faithfulness 评估，防止模型退化。
>
> 这套体系最大的价值是让 RAG 调优从**玄学变成了科学**。我可以清楚地告诉你，chunk_size 从 256 改到 512 让 Recall@5 提升了 12 个百分点，加混合检索又提升了 7 个百分点，加 Rerank 再提升 4 个百分点。每一步都有数据支撑。
>
> 另外我还发现一个关键点：**评测集本身也需要迭代**。随着业务发展，我会定期把新的 bad case 和用户高频查询加入评测集，确保评测覆盖最新场景。

### Q: "评测集怎么建？多大合适？"

> 我的经验是最少 20 条能跑起来，50 条比较稳，100 条以上统计显著性更好。构建方法我用三种组合：第一，从线上查询日志中挖掘高频真实问题；第二，用 LLM 基于知识库文档自动生成 QA pair；第三，请领域专家审核和补充高难度 case。
>
> 每条评测数据需要四个字段：question、ground_truth、relevant_chunk_ids、difficulty。其中 relevant_chunk_ids 最重要也最难标注——它决定了检索指标的准确性。我的做法是先让 LLM 初标，再人工复核，不一致的以人工为准。
>
> 评测集要定期更新，建议每月 review 一次，把新的 bad case 加进去。同时要注意 train-eval 隔离，评测集绝对不能用来调参或选 few-shot 样例。
