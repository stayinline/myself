## 一、搭建 Text2SQL Demo

### 1.1 完整代码实现

```python
"""text2sql.py - 自然语言 → SQL → 查询 → LLM 总结"""
import os
from openai import OpenAI
import clickhouse_connect  # 或 pymysql / psycopg2
import re
from typing import Dict, List, Optional, Tuple

client = OpenAI()


class Text2SQL:
    """安全的 Text2SQL 管线"""

    # 禁止的 SQL 关键词（安全白名单）
    BLOCKED_KEYWORDS = [
        "DROP", "INSERT", "UPDATE", "DELETE", "ALTER",
        "TRUNCATE", "CREATE", "GRANT", "REVOKE", "EXEC",
        "EXECUTE", "XP_", "SP_", "SHUTDOWN",
    ]

    MAX_ROWS = 100       # 最大返回行数
    QUERY_TIMEOUT = 30   # 查询超时(秒)

    def __init__(self, schema_desc: str, db_config: dict):
        self.schema_desc = schema_desc
        self.db_config = db_config
        # 使用只读账号连接
        self.db_client = clickhouse_connect.get_client(
            host=db_config["host"],
            port=db_config["port"],
            username=db_config["readonly_user"],  # ⚠️ 只读账号
            password=db_config["readonly_password"],
            database=db_config["database"],
        )

    def nl_to_sql(self, question: str) -> Tuple[str, str]:
        """Step 1: 自然语言 → SQL"""
        prompt = f"""你是一个 SQL 专家。根据以下数据库 Schema 和用户问题，生成 ClickHouse SQL。

【Schema】
{self.schema_desc}

【规则】
1. 只生成 SELECT 语句，禁止任何写操作
2. 必须加 LIMIT {self.MAX_ROWS}
3. 使用中文别名让结果更易读
4. 如果无法确定字段，说明原因而不是猜测

【用户问题】
{question}

【SQL】"""

        resp = client.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.0,
        )
        sql = resp.choices[0].message.content.strip()
        # 提取 SQL（去掉 markdown code block）
        sql = re.sub(r'```(?:sql)?\n?', '', sql).strip('`\n ')
        return sql, "generated"

    def validate_sql(self, sql: str) -> Tuple[bool, str, str]:
        """Step 2: SQL 安全校验，返回 (是否通过, 消息, 修正后的SQL)"""
        sql_upper = sql.upper()

        # 检查禁止关键词
        for keyword in self.BLOCKED_KEYWORDS:
            if keyword in sql_upper:
                return False, f"安全拦截: 包含禁止关键词 '{keyword}'", sql

        # 检查是否有 LIMIT，没有则自动追加
        if "LIMIT" not in sql_upper:
            sql = sql.rstrip(";") + f"\nLIMIT {self.MAX_ROWS}"

        # 基本语法检查
        if not sql_upper.strip().startswith("SELECT"):
            return False, "仅允许 SELECT 查询", sql

        return True, "通过", sql

    def execute_sql(self, sql: str) -> Tuple[List[Dict], str]:
        """Step 3: 执行 SQL 并返回结果"""
        try:
            result = self.db_client.query(sql, settings={
                "max_execution_time": self.QUERY_TIMEOUT,
                "max_rows_to_read": 1000000,  # 防止全表扫描
            })
            rows = [dict(zip(result.column_names, row)) for row in result.result_rows]
            return rows, "success"
        except Exception as e:
            return [], f"执行错误: {str(e)}"

    def summarize(self, question: str, sql: str, data: List[Dict]) -> str:
        """Step 4: LLM 总结查询结果"""
        # 大结果集先做摘要
        if len(data) > 20:
            data_preview = data[:20]
            summary_note = f"\n（注：共 {len(data)} 条结果，以上展示前 20 条）"
        else:
            data_preview = data
            summary_note = ""

        prompt = f"""根据以下 SQL 查询结果回答用户问题。用自然语言总结关键信息。

用户问题: {question}
执行的 SQL: {sql}
查询结果: {data_preview}{summary_note}

请用简洁清晰的中文回答:"""

        resp = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.0,
        )
        return resp.choices[0].message.content

    def ask(self, question: str) -> Dict:
        """完整 Text2SQL 流程"""
        # Step 1: NL → SQL
        sql, _ = self.nl_to_sql(question)

        # Step 2: 安全校验
        is_valid, msg, sql = self.validate_sql(sql)
        if not is_valid:
            return {"question": question, "sql": sql, "error": msg, "answer": None}

        # Step 3: 执行
        data, exec_msg = self.execute_sql(sql)
        if not data and "错误" in exec_msg:
            return {"question": question, "sql": sql, "error": exec_msg, "answer": None}

        # Step 4: 总结
        answer = self.summarize(question, sql, data)

        return {
            "question": question,
            "sql": sql,
            "row_count": len(data),
            "answer": answer,
            "raw_data": data[:5],  # 只保留前5条用于调试
        }
```

"""router.py - 意图路由"""

ROUTE_PROMPT = """判断以下用户问题应该使用哪种查询方式:
- "sql": 需要查数据库获取精确数值、统计、排名、趋势
- "vector": 需要语义搜索、相似查找、模糊匹配
- "both": 既需要精确数据又需要语义信息

只回复 sql / vector / both 其中一个词。

用户问题: {question}"""

def route_question(question: str) -> str:
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": ROUTE_PROMPT.format(question=question)}],
        temperature=0.0,
        max_tokens=5,
    )
    return resp.choices[0].message.content.strip().lower()


# 测试
print(route_question("上季度华东销售额"))      # → sql
print(route_question("有没有关于电池问题的反馈"))  # → vector
print(route_question("卖得最好的产品和它的差评"))   # → both

## 三、安全防护详解

### 3.1 四层防护代码

```python
"""security.py - Text2SQL 安全加固"""

class SQLSecurityGuard:
    """SQL 安全守卫"""

    READONLY_KEYWORDS = {"SELECT", "WITH", "FROM", "WHERE", "JOIN",
                         "GROUP", "ORDER", "HAVING", "LIMIT", "UNION"}
    BLOCKED_KEYWORDS = {"DROP", "INSERT", "UPDATE", "DELETE", "ALTER",
                        "TRUNCATE", "CREATE", "GRANT", "REVOKE",
                        "EXEC", "CALL", "INTO OUTFILE", "LOAD_FILE"}

    @classmethod
    def check(cls, sql: str) -> dict:
        checks = {
            "is_select_only": True,
            "has_limit": True,
            "no_blocked_keywords": True,
            "no_subquery_write": True,
            "errors": [],
        }

        sql_upper = sql.upper()

        # 1. 必须以 SELECT/WITH 开头
        first_word = sql_upper.strip().split()[0]
        if first_word not in ("SELECT", "WITH"):
            checks["is_select_only"] = False
            checks["errors"].append(f"非法起始关键字: {first_word}")

        # 2. 检查禁止词
        for kw in cls.BLOCKED_KEYWORDS:
            # 用单词边界匹配避免误杀（如 UPDATE_TIME 不应被拦截）
            pattern = rf'\b{kw}\b'
            if re.search(pattern, sql_upper):
                checks["no_blocked_keywords"] = False
                checks["errors"].append(f"包含禁止关键词: {kw}")

        # 3. 检查 LIMIT
        if "LIMIT" not in sql_upper:
            checks["has_limit"] = False
            checks["errors"].append("缺少 LIMIT 子句")

        return checks

    @classmethod
    def sanitize(cls, sql: str, max_rows: int = 100) -> str:
        """自动修复常见安全问题"""
        sql = sql.strip().rstrip(";")

        # 自动加 LIMIT
        if "LIMIT" not in sql.upper():
            sql += f"\nLIMIT {max_rows}"

        return sql
```

import json
from datetime import datetime

def log_text2sql_audit(
    user_id: str,

        "row_count": row_count,
        "latency_ms": latency_ms,

### 4.1 策略对比

| 策略 | 做法 | 适用场景 |
|------|------|---------|
| SQL 层聚合 | 在 SQL 中加 GROUP BY/SUM/COUNT | 统计分析类 |
| 截断 + 提示 | 取前 N 行 + 告知总数 | 列表展示类 |
| 两阶段查询 | 先 COUNT 再决定如何取数 | 不确定结果大小时 |
| 预计算物化视图 | 提前算好常用指标 | 高频查询 |

### 4.2 自动聚合逻辑

```python
def smart_summarize_data(data: List[Dict], question: str) -> str:
    """根据数据量和查询意图自动选择摘要策略"""
    if len(data) <= 10:
        return json.dumps(data, ensure_ascii=False)

    # 数据量大时，提供统计摘要而非原始数据
    numeric_cols = [k for k, v in data[0].items() if isinstance(v, (int, float))]

    summary = {"total_rows": len(data)}
    for col in numeric_cols:
        values = [row[col] for row in data if row[col] is not None]
        if values:
            summary[f"{col}_sum"] = sum(values)
            summary[f"{col}_avg"] = sum(values) / len(values)
            summary[f"{col}_min"] = min(values)
            summary[f"{col}_max"] = max(values)

    return json.dumps(summary, ensure_ascii=False)
```

## 五、面试话术

### Q: "用户用自然语言查数据库，你怎么做又准又安全？"

> 我做了一套完整的 Text2SQL 管线，分四步走：
>
> **第一步是 NL 转 SQL**。我把详细的 schema 描述（包括字段含义、枚举值、业务规则）作为 context 传给 LLM，让它生成 SQL。Schema 描述的质量是准确率的基础——这部分我发挥数据工程师的优势，写了非常详细的 table/column comment 和业务注释。
>
> **第二步是安全校验**。生成的 SQL 经过四层安全检查：是否纯 SELECT、有无禁止关键词、有无 LIMIT、资源配额限制。不通过的直接拦截并告知用户。
>
> **第三步是执行**。用专门的只读账号连库，设置查询超时和最大读取行数。所有执行都记审计日志。
>
> **第四步是结果总结**。小结果集直接给 LLM 总结；大结果集先在代码层做聚合统计，再把摘要喂给 LLM，避免 context 爆炸。
>
> 另外我还做了**意图路由**：精确查询走 Text2SQL，模糊语义走向量检索，混合需求并行执行后合并。这样覆盖了从精确到模糊的全谱系查询需求。
>
> 安全方面最关键的是：**永远不要用有写权限的账号执行 LLM 生成的 SQL**。这一条就杜绝了最严重的安全风险。

### Q: "Text2SQL 生成的 SQL 错了怎么办？"

> 我从三个层面应对：
>
> **预防层**：写好 schema description 和 few-shot examples。我发现加上 3-5 条典型的 NL-SQL pair 作为示例，准确率能从 70% 提升到 90%+。这些样例我从历史查询日志里挑，选覆盖各种 JOIN/GROUP BY/窗口函数的典型 case。
>
> **校验层**：对生成的 SQL 做语法检查和 EXPLAIN 分析。如果 EXPLAIN 显示全表扫描或者预估读取行数过大，自动拒绝并要求 LLM 重新生成。
>
> **反馈层**：如果执行报错，把错误信息连同原 SQL 一起送回 LLM 让它修正，最多重试 2 次。同时把所有 bad case 收集起来，定期更新 few-shot 样例和 schema 描述。这是一个持续改进的闭环。
