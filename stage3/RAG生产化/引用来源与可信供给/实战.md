## 一、为 Chunk 添加元数据 Schema

### 1.1 定义元数据模型

```python
"""chunk_metadata.py - Chunk 元数据标准 Schema"""
from dataclasses import dataclass, field, asdict
from datetime import datetime
from typing import List, Optional
import json
import hashlib


@dataclass
class ChunkMetadata:
    """每个 chunk 必须携带的元数据"""
    chunk_id: str               # 全局唯一 ID
    doc_id: str                 # 所属文档 ID
    source_url: str             # 原始文档 URL
    page_number: Optional[int]  # 页码（PDF/Word 场景）
    section_title: str          # 章节标题
    doc_version: str            # 文档版本
    created_at: str             # 创建时间 ISO 格式
    updated_at: str             # 更新时间 ISO 格式
    source_system: str          # 数据来源系统
    author: Optional[str]       # 内容作者
    tags: List[str]             # 业务标签
    access_level: str           # 访问权限: public/internal/confidential

    def to_dict(self) -> dict:
        return asdict(self)

    def to_json(self) -> str:
        return json.dumps(self.to_dict(), ensure_ascii=False)


def generate_chunk_id(doc_id: str, chunk_index: int, version: str) -> str:
    """生成确定性 chunk_id，相同输入产生相同 ID（便于去重和更新）"""
    raw = f"{doc_id}:{version}:{chunk_index}"
    hash_suffix = hashlib.md5(raw.encode()).hexdigest()[:8]
    return f"{doc_id}_v{version}_c{chunk_index:04d}_{hash_suffix}"


# 示例：为一个 chunk 创建元数据
meta = ChunkMetadata(
    chunk_id=generate_chunk_id("hr_leave_policy", 3, "3.2"),
    doc_id="hr_leave_policy",
    source_url="https://wiki.internal/hr/leave-policy-v3.2",
    page_number=12,
    section_title="第三章 年假规则",
    doc_version="3.2",
    created_at="2024-03-15T10:30:00Z",
    updated_at="2024-06-01T14:20:00Z",
    source_system="Confluence",
    author="张三 (HRBP)",
    tags=["hr", "leave", "annual-leave"],
    access_level="internal",
)
print(meta.to_json())
```

### 1.2 向量化时绑定元数据

```python
"""embedding_with_meta.py - Embedding 入库时绑定元数据"""
from pymilvus import Collection, DataType, FieldSchema, CollectionSchema

# Milvus Collection Schema：向量 + 元数据
fields = [
    FieldSchema(name="chunk_id", dtype=DataType.VARCHAR, is_primary=True, max_length=128),
    FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=1536),
    FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=65535),
    # 元数据字段
    FieldSchema(name="doc_id", dtype=DataType.VARCHAR, max_length=128),
    FieldSchema(name="source_url", dtype=DataType.VARCHAR, max_length=512),
    FieldSchema(name="section_title", dtype=DataType.VARCHAR, max_length=256),
    FieldSchema(name="doc_version", dtype=DataType.VARCHAR, max_length=32),
    FieldSchema(name="created_at", dtype=DataType.VARCHAR, max_length=32),
    FieldSchema(name="source_system", dtype=DataType.VARCHAR, max_length=64),
    FieldSchema(name="access_level", dtype=DataType.VARCHAR, max_length=16),
]

schema = CollectionSchema(fields=fields, description="RAG chunks with metadata")
collection = Collection(name="rag_chunks", schema=schema)

# 插入数据时同时写入向量和元数据
def insert_chunks(collection, chunks: list, embeddings: list, metadatas: list):
    data = [
        [m.chunk_id for m in metadatas],       # chunk_id
        embeddings,                              # embedding vectors
        [c for c in chunks],                     # content text
        [m.doc_id for m in metadatas],           # doc_id
        [m.source_url for m in metadatas],       # source_url
        [m.section_title for m in metadatas],    # section_title
        [m.doc_version for m in metadatas],      # doc_version
        [m.created_at for m in metadatas],       # created_at
        [m.source_system for m in metadatas],    # source_system
        [m.access_level for m in metadatas],     # access_level
    ]
    collection.insert(data)
    collection.flush()
```

## 二、检索时携带来源信息

### 2.1 检索返回结构化结果

```python
"""retrieve_with_source.py"""

def retrieve_with_metadata(collection, query_embedding, top_k=5,
                           access_filter=None):
    """检索并返回带完整元数据的结果"""
    search_params = {"metric_type": "COSINE", "params": {"nprobe": 10}}

    # 可选：按权限过滤
    expr = None
    if access_filter:
        expr = f'access_level == "{access_filter}"'

    results = collection.search(
        data=[query_embedding],
        anns_field="embedding",
        param=search_params,
        limit=top_k,
        expr=expr,
        output_fields=[
            "content", "doc_id", "source_url",
            "section_title", "doc_version", "created_at", "source_system"
        ],
    )

    formatted = []
    for hit in results[0]:
        formatted.append({
            "rank": hit.rank + 1,
            "score": hit.score,
            "content": hit.entity.get("content"),
            "source": {
                "doc_id": hit.entity.get("doc_id"),
                "url": hit.entity.get("source_url"),
                "section": hit.entity.get("section_title"),
                "version": hit.entity.get("doc_version"),
                "system": hit.entity.get("source_system"),
                "updated": hit.entity.get("created_at"),
            }
        })
    return formatted
```

### 2.2 构造带引用的 Prompt

```python
"""prompt_with_citations.py"""

def build_citation_prompt(query: str, retrieved: list) -> str:
    """构建带引用标注要求的 Prompt"""
    context_parts = []
    for i, item in enumerate(retrieved, 1):
        src = item["source"]
        source_label = f'{src["system"]} - {src["section"]} (v{src["version"]})'
        context_parts.append(
            f"[{i}] {item['content']}\n    📎 来源: {source_label} | {src['url']}"
        )

    context_block = "\n\n".join(context_parts)

    prompt = f"""你是一个企业知识库助手。请严格遵守以下规则：

【引用规则】
1. 仅基于下方【参考资料】回答，不得使用自身知识。
2. 每个事实性陈述后标注引用编号 [n]。
3. 多个来源支持同一观点时标注所有编号，如 [1][3]。
4. 资料中无相关信息时回答"根据提供的资料无法回答此问题"。
5. 绝对不得编造不存在的引用编号。

【参考资料】
{context_block}

【用户问题】
{query}

【回答】"""
    return prompt
```

## 三、引用验证后处理

### 3.1 完整验证管线

```python
"""citation_validator.py"""
import re
from typing import Dict, List, Tuple


class CitationValidator:
    """验证并清洗模型输出中的引用"""

    def __init__(self, num_sources: int):
        self.valid_ids = set(range(1, num_sources + 1))

    def validate(self, answer: str) -> Dict:
        """提取并验证所有引用"""
        cited = set(int(m) for m in re.findall(r'\[(\d+)\]', answer))
        fabricated = cited - self.valid_ids
        valid = cited & self.valid_ids

        return {
            "original_answer": answer,
            "cited_ids": sorted(cited),
            "valid_ids": sorted(valid),
            "fabricated_ids": sorted(fabricated),
            "is_clean": len(fabricated) == 0,
        }

    def clean(self, answer: str) -> Tuple[str, Dict]:
        """移除编造的引用，保留合法引用"""
        validation = self.validate(answer)
        cleaned = answer

        # 移除编造的引用标记
        for fid in validation["fabricated_ids"]:
            cleaned = re.sub(rf'\[{fid}\]', '', cleaned)

        # 清理多余空格
        cleaned = re.sub(r'\s+', ' ', cleaned).strip()

        return cleaned, validation


# 使用示例
validator = CitationValidator(num_sources=5)

answer = "年假余额为5天 [1]，超过3年未使用的年假将自动作废 [3][7]。"
cleaned, report = validator.clean(answer)

print(f"原文: {answer}")
print(f"清洗: {cleaned}")
print(f"报告: {report}")
# fabricated_ids: [7] → 被移除
# 输出: "年假余额为5天 [1]，超过3年未使用的年假将自动作废 [3]。"
```

### 3.2 前端渲染引用

```python
def render_answer_with_sources(cleaned_answer: str, retrieved: list) -> str:
    """将引用编号渲染为可点击的来源链接"""
    result = cleaned_answer

    for i, item in enumerate(retrieved, 1):
        src = item["source"]
        link_text = f'{src["system"]} - {src["section"]}'
        replacement = f'<a href="{src["url"]}" title="v{src["version"]}">{link_text}</a>'
        result = result.replace(f'[{i}]', f'<sup>{replacement}</sup>')

    return result
```

## 五、常见坑与应对

### 5.1 坑位清单

| 坑 | 表现 | 原因 | 解法 |
|----|------|------|------|
| 模型编造引用 | `[7]` 但只有 5 个来源 | LLM 概率生成本质 | 后处理验证 + Prompt 强约束 |
| 元数据缺失 | chunk 无法追溯 | 入库流程不规范 | 入库前校验，缺 meta 拒绝写入 |
| 引用位置错误 | 张冠李戴 | 模型混淆了相似 chunk | Rerank + 减少 top-k |
| 版本混乱 | 引用了旧版本文档 | 更新后旧 chunk 未标记 | 版本化存储 + deprecated 标记 |
| 权限泄露 | 回答了机密内容 | 未按 access_level 过滤 | 检索时加 expr 过滤条件 |
| URL 失效 | 点击引用打不开 | 源文档迁移/删除 | 定期巡检 + 快照备份 |

### 5.2 元数据质量保障

```python
def validate_chunk_metadata(meta: ChunkMetadata) -> bool:
    """入库前元数据完整性校验"""
    required_fields = ["chunk_id", "doc_id", "source_url", "doc_version",
                       "created_at", "source_system", "access_level"]
    for f in required_fields:
        if not getattr(meta, f):
            raise ValueError(f"元数据缺少必填字段: {f}")

    # URL 格式校验
    if not meta.source_url.startswith(("http", "/", "s3://", "hdfs://")):
        raise ValueError(f"source_url 格式异常: {meta.source_url}")

    # access_level 枚举校验
    valid_levels = {"public", "internal", "confidential", "restricted"}
    if meta.access_level not in valid_levels:
        raise ValueError(f"非法 access_level: {meta.access_level}")

    return True
```

## 六、面试话术

### Q: "怎么让 RAG 的答案可信、可追溯到原文？"

> 我做了一套三层机制：
>
> **第一层是数据侧**。我在 chunking pipeline 里设计了标准化的元数据 schema，每个 chunk 入库时必须带上 doc_id、source_url、page_number、version、timestamp、author、access_level 等字段。这些元数据和向量一起存入 Milvus，作为后续检索过滤和引用展示的基础。入库前有自动校验，缺字段的 chunk 直接拒绝写入。
>
> **第二层是 Prompt 侧**。我在 system prompt 里明确要求模型对每个事实陈述标注 [n] 引用编号，并且强调"不得编造不存在的编号"。同时在传入的参考资料中，每条都附带来源摘要信息，让模型有东西可引。
>
> **第三层是验证侧**。模型输出后我会跑一个 CitationValidator，用正则提取所有引用编号，和实际传入的 chunk 数量做比对。发现编造的引用就自动移除或标红。对于高合规场景，还会让模型以 JSON 结构化输出，方便程序化逐条验证。
>
> 另外我还建了数据血缘表，记录从原始文档到 chunk 到向量索引的全链路关系。这样出了问题可以从回答一路追溯到具体哪个文档的哪个版本哪个段落，满足审计要求。
>
> 这套组合我叫做**可信数据供给**，是企业级 RAG 和普通 demo 的核心区别。

### Q: "你们怎么处理敏感数据的权限控制？"

> 在 chunk 元数据里有 access_level 字段，分为 public / internal / confidential / restricted 四级。检索时在 Milvus 的 search 接口加 expr 过滤条件，比如当前用户是 internal 级别，就只检索 access_level in ('public', 'internal') 的 chunk。这样从检索层面就杜绝了越权访问，而不是靠生成侧去过滤——后者不安全，因为敏感内容已经进入 prompt 了。
