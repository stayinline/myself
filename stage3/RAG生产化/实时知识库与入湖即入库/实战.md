```python
"""flink_rag_pipeline.py - Flink 实时 RAG 数据处理管线"""
# 注: 实际生产用 Java/Scala Flink API，此处用 Python 伪代码展示逻辑

from pyflink.datastream import StreamExecutionEnvironment
from pyflink.datastream.connectors.kafka import KafkaSource, KafkaSink
import hashlib
import json


class RAGChunkingPipeline:
    """
    Flink 管线核心逻辑:
    Kafka(CDC) → Parse → Chunk → Dedup → Embed(Async) → DualWrite(Paimon+Milvus)
    """

    def __init__(self):
        self.env = StreamExecutionEnvironment.get_execution_environment()
        self.env.set_parallelism(4)

    def build_pipeline(self):
        # Step 1: 从 Kafka 读取变更事件
        source = KafkaSource.builder() \
            .set_bootstrap_servers("localhost:9092") \
            .set_topics("knowledge_changes") \
            .set_group_id("rag-pipeline-group") \
            .build()

        stream = self.env.from_source(source, WatermarkStrategy.no_watermarks(), "CDC Source")

        # Step 2: 解析 + Chunking
        chunked_stream = stream.map(parse_and_chunk)

        # Step 3: 内容去重（基于 content hash）
        deduped_stream = chunked_stream.key_by(lambda c: c["content_hash"]) \
            .process(ContentDedupFunction())

        # Step 4: Async I/O 调用 Embedding API
        embedded_stream = AsyncDataStream.ordered_wait(
            deduped_stream,
            EmbeddingAsyncFunction(),
            timeout_ms=10000,
            capacity=100,       # 最大并发
        )

        # Step 5: 双写 Paimon + Milvus
        embedded_stream.add_sink(PaimonSink("rag_chunks"))
        embedded_stream.add_sink(MilvusSink("rag_vectors"))

        return self.env


def parse_and_chunk(event: dict) -> list:
    """解析变更事件并做 Chunking"""
    if event["event_type"] == "DELETE":
        return [{"type": "DELETE", "doc_id": event["doc_id"],
                 "version": event["metadata"]["version"]}]

    content = event["content"]
    chunks = smart_chunk(content, chunk_size=512, overlap=64)

    results = []

            "doc_id": event["doc_id"],

            "content": chunk_text,
            "content_hash": content_hash,
            "metadata": event["metadata"],
            "source_system": event["source_system"],
            "timestamp": event["timestamp"],
        })
    return results
```

### 2.2 Smart Chunking 实现

```python
"""chunking.py - 智能分块策略"""
import re
from typing import List


def smart_chunk(text: str, chunk_size: int = 512, overlap: int = 64) -> List[str]:
    """
    优先按段落切分，超长段落再按句子切分
    """
    # 第一级：按段落分割
    paragraphs = text.split("\n\n")

    chunks = []
    current_chunk = ""

    for para in paragraphs:
        para = para.strip()
        if not para:
            continue

        if len(current_chunk) + len(para) + 1 <= chunk_size:
            current_chunk += ("\n\n" + para) if current_chunk else para
        else:
            # 当前 chunk 已满，保存并开始新的
            if current_chunk:
                chunks.append(current_chunk)

            # 如果单个段落就超过 chunk_size，按句子切分
            if len(para) > chunk_size:
                sentences = re.split(r'(?<=[。！？\.\!\?])', para)
                current_chunk = ""
                for sent in sentences:
                    if len(current_chunk) + len(sent) <= chunk_size:
                        current_chunk += sent
                    else:
                        if current_chunk:
                            chunks.append(current_chunk)
                        current_chunk = sent
            else:
                current_chunk = para

    if current_chunk:
        chunks.append(current_chunk)

    # 添加 overlap
    if overlap > 0 and len(chunks) > 1:
        overlapped = [chunks[0]]
        for i in range(1, len(chunks)):
            prev_tail = chunks[i-1][-overlap:]
            overlapped.append(prev_tail + chunks[i])
        chunks = overlapped

    return chunks
```

## 三、向量库增删改一致性处理

### 3.1 软删除 + 版本过滤

```python
"""vector_consistency.py"""
from pymilvus import Collection


class VectorStoreManager:
    """管理向量库的增量更新"""

    def __init__(self, collection: Collection):
        self.collection = collection

    def handle_insert(self, chunks: list, embeddings: list):
        """新增: 直接插入"""
        data = self._prepare_data(chunks, embeddings, deprecated=False)
        self.collection.insert(data)

    def handle_update(self, doc_id: str, new_version: str,
                      new_chunks: list, new_embeddings: list):
        """修改: 标记旧版本 + 插入新版本"""
        # Step 1: 软删除旧版本
        self.collection.delete(
            expr=f'doc_id == "{doc_id}" AND deprecated == false'
        )
        # 注意: Milvus delete 是异步的，但因为我们用 deprecated 字段过滤，
        # 即使物理删除未完成，检索也不会返回旧数据

        # Step 2: 插入新版本
        data = self._prepare_data(new_chunks, new_embeddings, deprecated=False)
        self.collection.insert(data)

    def handle_delete(self, doc_id: str):
        """删除: 软删除所有关联 chunks"""
        self.collection.delete(
            expr=f'doc_id == "{doc_id}" AND deprecated == false'
        )

    def search(self, query_embedding, top_k=5, access_level=None):
        """检索时自动过滤已废弃的 chunks"""
        expr = 'deprecated == false'
        if access_level:
            expr += f' AND access_level == "{access_level}"'

        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k,
            expr=expr,
            output_fields=["content", "doc_id", "source_url", "doc_version"],
        )
        return results

    def cleanup_deprecated(self, days_threshold: int = 30):
        """定期物理清理过期的 deprecated chunks"""
        cutoff = f"now() - {days_threshold}d"
        self.collection.delete(
            expr=f'deprecated == true AND updated_at < "{cutoff}"'
        )
```

### 3.2 连接 D47（数据一致性设计模式）

```
向量库一致性保障 checklist:
□ 每个 chunk 有唯一 chunk_id (primary key)
□ 每个 chunk 有 doc_id + version 用于关联
□ 修改操作 = 软删旧版 + 插入新版
□ 删除操作 = 软删 (deprecated=true)
□ 检索时过滤 deprecated=false
□ 定时任务物理清理过期 deprecated 记录
□ 写入失败有重试机制 (最多3次)
□ 写入异常有告警 (Prometheus counter)
□ Paimon 和 Milvus 写入独立事务 (最终一致)
□ 血缘表记录每次变更的操作类型和时间
```

## 四、Embedding 调用的成本控制

### 4.1 限流器实现

```python
"""rate_limiter.py - 令牌桶限流"""
import asyncio
import time


class TokenBucketRateLimiter:
    """令牌桶限流器，控制 Embedding API 调用速率"""

    def __init__(self, rate: float, capacity: int):
        """
        rate: 每秒产生的令牌数
        capacity: 桶容量（突发量）
        """
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_refill = time.monotonic()
        self._lock = asyncio.Lock()

    async def acquire(self, tokens: int = 1):
        async with self._lock:
            now = time.monotonic()
            elapsed = now - self.last_refill
            self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
            self.last_refill = now

            if self.tokens >= tokens:
                self.tokens -= tokens
                return True
            else:
                wait_time = (tokens - self.tokens) / self.rate
                await asyncio.sleep(wait_time)
                self.tokens = 0
                self.last_refill = time.monotonic()
                return True


# 使用: 限制为每分钟 1000 次调用
limiter = TokenBucketRateLimiter(rate=16.7, capacity=50)

async def embed_with_limit(texts: list):
    await limiter.acquire(len(texts))
    return await call_embedding_api(texts)
```

### 4.3 重试策略

```python
"""retry_strategy.py"""
import asyncio
import random
from openai import OpenAI, RateLimitError, APIError

client = OpenAI()


async def embed_with_retry(texts: list, max_retries: int = 3) -> list:
    """带指数退避重试的 Embedding 调用"""
    for attempt in range(max_retries + 1):
        try:
            resp = client.embeddings.create(
                input=texts,
                model="text-embedding-3-small",
            )
            return [d.embedding for d in resp.data]

        except RateLimitError:
            if attempt == max_retries:
                raise
            wait = (2 ** attempt) + random.uniform(0, 1)
            print(f"Rate limited, retrying in {wait:.1f}s (attempt {attempt+1})")
            await asyncio.sleep(wait)

        except APIError as e:
            if attempt == max_retries:
                raise
            wait = (2 ** attempt) + random.uniform(0, 1)
            print(f"API error: {e}, retrying in {wait:.1f}s")
            await asyncio.sleep(wait)
```

## 六、面试话术

### Q: "介绍一下你的自动化 Embedding 数据流水线？"

> 这是我方法论中的旗舰项目——**入湖即入库**。它解决了企业 RAG 知识库更新滞后的核心痛点。
>
> 整套 pipeline 是这样的：数据源发生变更时，通过 CDC 或 Kafka 消息实时捕获变更事件。Flink 流式处理引擎收到事件后，做三件事：第一，按文档结构智能 Chunking；第二，通过 Async I/O 批量调用 Embedding API，同时做了内容哈希去重避免重复调用；第三，将结果双写到 Paimon（存明细和血缘）和 Milvus（存向量和元数据）。
>
> 对于三种变更状态我分别处理：新增直接写入；修改采用软删除加版本过滤——旧 chunk 标记 deprecated，检索时自动排除，避免了向量库 DELETE 操作的异步一致性问题；删除同样走软删除。
>
> 成本方面，通过 Batch Embedding、内容去重和令牌桶限流，把 Embedding 调用成本控制在合理范围。实测一个日均 500 次变更的中型知识库，每天 Embedding 成本不到 ¥0.6。
>
> 最终效果是**分钟级的知识库更新**——用户问到的永远是最新信息。这套方案展示了数据工程师从 CDC 采集、Flink 流处理、数据湖存储到 AI 服务集成的全栈能力。

### Q: "Flink Async I/O 调用 Embedding 有什么注意事项？"

> 三个关键点：
>
> **第一是 capacity 设置**。要根据 Embedding API 的 rate limit 来设。比如 API 限制每分钟 1000 次，那 capacity 设 50-100 比较安全，太高会被限流导致大量重试。
>
> **第二是 ordered 参数**。我建议设为 true，保证输出顺序和输入一致。虽然 unordered 吞吐更高，但在需要因果序的场景（比如同一文档的多次更新）会导致版本错乱。
>
> **第三是容错**。Async I/O 中的失败需要有重试机制。我用指数退避重试，最多 3 次。超过重试次数的记录写入死信队列，后续人工处理。同时所有 API 调用都埋了 Prometheus metrics，监控成功率和延迟分布。
>
> 另外还有一个工程细节：Embedding API 通常支持 batch 调用，我在 AsyncFunction 里做了一个微型 buffer，攒够 10 条或等 100ms 就发一批，比逐条调用效率高 5-8 倍。
