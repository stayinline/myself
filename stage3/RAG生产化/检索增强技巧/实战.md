## 一、搭建混合检索 + RRF + Rerank 完整管线

### 1.1 整体架构代码

```python
"""hybrid_retriever.py - 向量 + BM25 + RRF + Rerank 完整管线"""
import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass


@dataclass
class SearchResult:
    doc_id: str
    content: str
    score: float
    source: str  # "vector" / "bm25" / "rerank"


class HybridRetriever:
    def __init__(self, vector_store, bm25_index, reranker=None):
        self.vector_store = vector_store   # Milvus/Qdrant 等
        self.bm25_index = bm25_index       # ES/BM25Retriever
        self.reranker = reranker           # Cross-Encoder (可选)

    def search(
        self,
        query: str,
        vector_top_k: int = 50,
        bm25_top_k: int = 50,
        final_top_k: int = 5,
        use_rerank: bool = True,
        rrf_k: int = 60,
    ) -> List[SearchResult]:
        """完整的混合检索管线"""

        # Step 1: 双路并行召回
        vector_results = self._vector_search(query, vector_top_k)
        bm25_results = self._bm25_search(query, bm25_top_k)

        # Step 2: RRF 融合
        fused = self._rrf_fusion(
            [vector_results, bm25_results],
            k=rrf_k,
        )

        # Step 3: 可选 Rerank
        if use_rerank and self.reranker:
            # 取融合后 top-N 做精排
            candidates = fused[:min(len(fused), 20)]
            reranked = self._rerank(query, candidates)
            return reranked[:final_top_k]

        return fused[:final_top_k]

    def _vector_search(self, query: str, top_k: int) -> List[Tuple[str, str, float]]:
        """向量检索路"""
        # 调用 Embedding API
        embedding = get_embedding(query)
        results = self.vector_store.search(embedding, limit=top_k)
        return [(r.doc_id, r.content, r.score) for r in results]

    def _bm25_search(self, query: str, top_k: int) -> List[Tuple[str, str, float]]:
        """BM25 关键词检索路"""
        results = self.bm25_index.search(query, top_k=top_k)
        return [(r.doc_id, r.content, r.score) for r in results]

    @staticmethod
    def _rrf_fusion(
        rankings: List[List[Tuple[str, str, float]]],
        k: int = 60,
    ) -> List[SearchResult]:
        """Reciprocal Rank Fusion"""
        scores: Dict[str, float] = {}
        contents: Dict[str, str] = {}

        for ranking in rankings:
            for rank, (doc_id, content, _) in enumerate(ranking):
                if doc_id not in scores:
                    scores[doc_id] = 0.0
                scores[doc_id] += 1.0 / (k + rank + 1)
                contents[doc_id] = content

        sorted_docs = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        return [
            SearchResult(doc_id=did, content=contents[did], score=sc, source="rrf")
            for did, sc in sorted_docs
        ]

    def _rerank(self, query: str, candidates: List[SearchResult]) -> List[SearchResult]:
        """Cross-Encoder 精排"""
        pairs = [[query, c.content] for c in candidates]
        rerank_scores = self.reranker.predict(pairs)

        for candidate, score in zip(candidates, rerank_scores):
            candidate.score = float(score)
            candidate.source = "rerank"

        candidates.sort(key=lambda x: x.score, reverse=True)
        return candidates
```

### 1.2 BM25 组件接入（基于 Elasticsearch）

```python
"""bm25_es.py - 接入 ES 作为 BM25 检索路"""
from elasticsearch import Elasticsearch


class ESBM25Index:
    def __init__(self, es_url="http://localhost:9200", index_name="rag_chunks"):
        self.es = Elasticsearch(es_url)
        self.index_name = index_name

    def index_chunks(self, chunks: List[Dict]):
        """批量写入 chunk 到 ES"""
        actions = []
        for chunk in chunks:
            actions.append({
                "_index": self.index_name,
                "_id": chunk["chunk_id"],
                "_source": {
                    "content": chunk["content"],
                    "doc_id": chunk["doc_id"],
                    "section_title": chunk.get("section_title", ""),
                }
            })
        # bulk 写入
        from elasticsearch.helpers import bulk
        bulk(self.es, actions)

    def search(self, query: str, top_k: int = 50) -> List[Dict]:
        """BM25 检索"""
        body = {
            "query": {
                "multi_match": {
                    "query": query,
                    "fields": ["content^2", "section_title"],
                    "type": "best_fields",
                }
            },
            "size": top_k,
        }
        resp = self.es.search(index=self.index_name, body=body)
        return [
            {
                "doc_id": hit["_id"],
                "content": hit["_source"]["content"],
                "score": hit["_score"],
            }
            for hit in resp["hits"]["hits"]
        ]
```

## 二、Recall@K 对比评测

### 2.1 评测脚本

```python
"""eval_recall.py - 量化检索增强效果"""
from typing import List, Dict


def evaluate_recall(
    retriever,
    eval_set: List[Dict],
    k_values: List[int] = [3, 5, 10],
) -> Dict:
    """
    eval_set: [{"query": "...", "relevant_doc_ids": ["id1", "id2"]}, ...]
    """
    results = {}

    for k in k_values:
        hits = 0
        total = len(eval_set)

        for item in eval_set:
            retrieved = retriever.search(item["query"], final_top_k=k)
            retrieved_ids = set(r.doc_id for r in retrieved)
            relevant_ids = set(item["relevant_doc_ids"])

            # Hit: 至少一个相关文档被召回
            if retrieved_ids & relevant_ids:
                hits += 1

        results[f"Recall@{k}"] = hits / total

    return results


# 使用示例
eval_set = [
    {"query": "退货政策是什么", "relevant_doc_ids": ["chunk_001"]},
    {"query": "HTTP 502 错误怎么处理", "relevant_doc_ids": ["chunk_042", "chunk_043"]},
    {"query": "Q3华东区销售额", "relevant_doc_ids": ["chunk_087"]},
    # ... 20-50 条评测数据
]

# 对比纯向量 vs 混合检索
print("=== 纯向量检索 ===")
print(evaluate_recall(vector_only_retriever, eval_set))

print("\n=== 向量 + BM25 + RRF ===")
print(evaluate_recall(hybrid_retriever, eval_set, use_rerank=False))

print("\n=== 向量 + BM25 + RRF + Rerank ===")
print(evaluate_recall(hybrid_retriever, eval_set, use_rerank=True))
```

```python
def weighted_fusion(vector_results, bm25_results, alpha=0.7):
    """
    alpha: 向量权重 (1-alpha 为 BM25 权重)
    需要先对两路分数做 min-max 归一化
    """
    # 归一化
    v_scores = normalize([r.score for r in vector_results])
    b_scores = normalize([r.score for r in bm25_results])

    # 合并
    all_docs = {}
    for doc, score in zip(vector_results, v_scores):
        all_docs[doc.doc_id] = {"content": doc.content, "score": alpha * score}
    for doc, score in zip(bm25_results, b_scores):
        if doc.doc_id in all_docs:
            all_docs[doc.doc_id]["score"] += (1 - alpha) * score
        else:
            all_docs[doc.doc_id] = {"content": doc.content, "score": (1 - alpha) * score}

    return sorted(all_docs.items(), key=lambda x: x[1]["score"], reverse=True)
```

```python
from FlagEmbedding import FlagReranker

reranker = FlagReranker(
    'BAAI/bge-reranker-v2-m3',
    use_fp16=True,  # GPU 推理用 fp16 加速

def rerank_pairs(query: str, documents: list[str]) -> list[float]:
    pairs = [[query, doc] for doc in documents]
    scores = reranker.compute_score(pairs, normalize=True)
    return scores if isinstance(scores, list) else [scores]
```

## 六、面试话术

### Q: "纯向量召回不准，你有哪些提升手段？"

> 我的增强方案是分层的，按 ROI 排序：
>
> **首先加 BM25 混合检索**。向量擅长语义但对精确术语弱，BM25 正好互补。我用 Elasticsearch 做 BM25 路——这对数据工程师来说是现有技术栈的自然延伸。两路结果用 RRF 融合，不需要分数归一化，简单有效。这一步通常能把 Recall@5 提升 15-20 个百分点。
>
> **然后加 Rerank**。粗召回 top-50，用 cross-encoder 精排取 top-5。我们用的 bge-reranker-v2-m3，GPU batch 推理 20 条约 120ms。这一步再提升 8-10 个百分点。
>
> **再看是否需要 Query 改写**。对于口语化严重的场景做同义词扩展；复合问题做子问题分解。HyDE 效果好但有额外延迟，我一般只在离线场景或低置信度 fallback 时使用。
>
> **所有调整都用评测集驱动**。我有一个 50 条标注数据的评测集，每次改完跑一遍 Recall@K，确认有提升才上线。最终我们从纯向量的 Recall@5=62% 做到了混合+Rerank 的 89%，延迟控制在 350ms 以内。

### Q: "RRF 和加权融合怎么选？"

> 大多数场景我推荐 RRF，因为它不需要对不同路的分数做归一化——向量分数和 BM25 分数量纲完全不同，归一化本身就会引入误差。RRF 只看排名不看绝对分数，鲁棒性更好。
>
> 加权融合适合你有明确的业务偏好时——比如你知道在某些场景下 BM25 比向量重要得多，可以给 BM25 更高权重。但这需要在评测集上 grid search 找最优 alpha，维护成本更高。
>
> 我的建议是先用 RRF 做 baseline，如果评测显示某一路明显被低估，再考虑切到加权融合。
